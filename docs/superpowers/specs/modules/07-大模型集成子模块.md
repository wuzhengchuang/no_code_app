# 子模块 07: 大模型集成子模块设计文档

## 概述

负责LLM抽象、提示词管理、代码生成请求。支持切换不同的大模型提供商，包括 Claude、OpenAI、Gemini 等，提供统一的接口进行代码生成、文档生成、对话交互等功能。

## LLM提供者抽象基类

```python
# llm/providers/base.py
from abc import ABC, abstractmethod
from typing import Dict, Any, List, Optional, AsyncGenerator
from llm.models.base import (
    LLMProviderType,
    Message,
    GenerationResult,
    LLMConfig
)

class LLMProvider(ABC):
    """大模型提供者抽象基类"""

    provider_type: LLMProviderType

    def __init__(self, config: LLMConfig):
        self.config = config
        self._validate_config()

    @abstractmethod
    def _validate_config(self) -> None:
        """验证配置"""
        pass

    @abstractmethod
    async def generate(
        self,
        prompt: str,
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> GenerationResult:
        """
        单次生成

        Args:
            prompt: 提示词
            max_tokens: 最大Token数
            temperature: 温度参数
            **kwargs: 其他参数

        Returns:
            GenerationResult: 生成结果
        """
        pass

    @abstractmethod
    async def chat(
        self,
        messages: List[Message],
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> GenerationResult:
        """
        对话交互

        Args:
            messages: 消息列表
            max_tokens: 最大Token数
            temperature: 温度参数
            **kwargs: 其他参数

        Returns:
            GenerationResult: 生成结果
        """
        pass

    @abstractmethod
    async def stream_chat(
        self,
        messages: List[Message],
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> AsyncGenerator[str, None]:
        """
        流式对话

        Args:
            messages: 消息列表
            max_tokens: 最大Token数
            temperature: 温度参数
            **kwargs: 其他参数

        Yields:
            str: 生成的内容片段
        """
        pass

    @abstractmethod
    def get_available_models(self) -> List[str]:
        """
        获取可用模型列表

        Returns:
            List[str]: 模型名称列表
        """
        pass

    @abstractmethod
    def count_tokens(self, text: str) -> int:
        """
        计算Token数

        Args:
            text: 文本

        Returns:
            int: Token数
        """
        pass

    @classmethod
    def get_default_model(cls) -> str:
        """
        获取默认模型

        Returns:
            str: 默认模型名称
        """
        pass
```

## Claude提供者实现

```python
# llm/providers/claude.py
import httpx
import time
from typing import Dict, Any, List, Optional, AsyncGenerator
from llm.providers.base import LLMProvider
from llm.models.base import (
    LLMProviderType,
    Message,
    GenerationResult,
    LLMConfig
)

class ClaudeProvider(LLMProvider):
    """Claude API实现"""

    provider_type = LLMProviderType.CLAUDE

    DEFAULT_MODEL = "claude-3-opus-20240229"

    AVAILABLE_MODELS = [
        "claude-3-opus-20240229",
        "claude-3-sonnet-20240229",
        "claude-3-haiku-20240307",
        "claude-2.1",
        "claude-2.0"
    ]

    def _validate_config(self) -> None:
        if not self.config.api_key:
            raise ValueError("API Key is required for Claude")

        if not self.config.model:
            self.config.model = self.DEFAULT_MODEL

    async def generate(
        self,
        prompt: str,
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> GenerationResult:
        start_time = time.time()

        messages = [Message(role="user", content=prompt)]

        return await self.chat(
            messages=messages,
            max_tokens=max_tokens,
            temperature=temperature,
            **kwargs
        )

    async def chat(
        self,
        messages: List[Message],
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> GenerationResult:
        start_time = time.time()

        api_base = self.config.api_base or "https://api.anthropic.com/v1"
        url = f"{api_base}/messages"

        headers = {
            "x-api-key": self.config.api_key,
            "anthropic-version": "2023-06-01",
            "content-type": "application/json"
        }

        formatted_messages = self._format_messages(messages)

        payload = {
            "model": self.config.model,
            "max_tokens": max_tokens or self.config.max_tokens,
            "messages": formatted_messages,
            "temperature": temperature if temperature is not None else self.config.temperature
        }

        if self.config.top_p is not None:
            payload["top_p"] = self.config.top_p

        try:
            async with httpx.AsyncClient(timeout=120.0) as client:
                response = await client.post(url, headers=headers, json=payload)
                response.raise_for_status()
                data = response.json()

            content = data["content"][0]["text"]
            usage = data.get("usage", {})

            return GenerationResult(
                success=True,
                content=content,
                prompt_tokens=usage.get("input_tokens"),
                completion_tokens=usage.get("output_tokens"),
                total_tokens=usage.get("input_tokens", 0) + usage.get("output_tokens", 0) if usage else None,
                model=data.get("model"),
                duration_ms=int((time.time() - start_time) * 1000),
                raw_response=data
            )

        except httpx.HTTPStatusError as e:
            error_data = e.response.json() if e.response.content else {}
            return GenerationResult(
                success=False,
                error=str(e),
                error_code=error_data.get("type", "http_error"),
                raw_response=error_data,
                duration_ms=int((time.time() - start_time) * 1000
            )

        except Exception as e:
            return GenerationResult(
                success=False,
                error=str(e),
                error_code="unknown_error",
                duration_ms=int((time.time() - start_time) * 1000
            )

    async def stream_chat(
        self,
        messages: List[Message],
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> AsyncGenerator[str, None]:
        api_base = self.config.api_base or "https://api.anthropic.com/v1"
        url = f"{api_base}/messages"

        headers = {
            "x-api-key": self.config.api_key,
            "anthropic-version": "2023-06-01",
            "content-type": "application/json"
        }

        formatted_messages = self._format_messages(messages)

        payload = {
            "model": self.config.model,
            "max_tokens": max_tokens or self.config.max_tokens,
            "messages": formatted_messages,
            "temperature": temperature if temperature is not None else self.config.temperature,
            "stream": True
        }

        try:
            async with httpx.AsyncClient(timeout=120.0) as client:
                async with client.stream("POST", url, headers=headers, json=payload) as response:
                    response.raise_for_status()
                    async for line in response.aiter_lines():
                        if line.startswith("data: "):
                            data = line[6:].strip()
                            if data:
                                import json
                                try:
                                    event_data = json.loads(data)
                                    if event_data.get("type") == "content_block_delta":
                                    delta = event_data.get("delta", {})
                                    if delta.get("type") == "text_delta":
                                        yield delta.get("text", "")
                                except json.JSONDecodeError:
                                    pass

        except Exception as e:
            yield f"\n[Error: {str(e)}]"

    def get_available_models(self) -> List[str]:
        return self.AVAILABLE_MODELS

    def count_tokens(self, text: str) -> int:
        import tiktoken
        try:
            encoding = tiktoken.get_encoding("cl100k_base")
            return len(encoding.encode(text))
        except:
            return len(text.split())

    @classmethod
    def get_default_model(cls) -> str:
        return cls.DEFAULT_MODEL

    def _format_messages(self, messages: List[Message]) -> List[Dict[str, Any]]:
        formatted = []
        for msg in messages:
            if msg.role in ["user", "assistant"]:
                formatted.append({
                    "role": msg.role,
                    "content": msg.content
                })
        return formatted

            return GenerationResult(

                content=content,
                prompt_tokens=usage.get("prompt_tokens"),

                total_tokens=usage.get("total_tokens"),

                error=str(e),
                error_code=error_data.get("error", {}).get("code", "http_error"),

                success=False,
                error=str(e),

            async with httpx.AsyncClient(timeout=120.0) as client:

                            data = line[6:].strip()
                            if data and data != "[DONE]":
                                import json
                                try:
                                    event_data = json.loads(data)
                                    delta = event_data.get("choices", [{}])[0].get("delta", {})
                                    if delta.get("content"):
                                        yield delta["content"]
                                except json.JSONDecodeError:
                                    pass

        except Exception as e:
            yield f"\n[Error: {str(e)}]"

    def get_available_models(self) -> List[str]:
        return self.AVAILABLE_MODELS

    def count_tokens(self, text: str) -> int:
        try:
            import tiktoken
            encoding = tiktoken.encoding_for_model(self.config.model)
            return len(encoding.encode(text))
        except:
            return len(text.split())

    @classmethod
    def get_default_model(cls) -> str:
        return cls.DEFAULT_MODEL

    def _format_messages(self, messages: List[Message]) -> List[Dict[str, Any]]:
        formatted = []
        for msg in messages:
            role = msg.role
            if role == "assistant":
                role = "assistant"
            elif role not in ["system", "user", "assistant"]:
                role = "user"

            formatted.append({
                "role": role,
                "content": msg.content
            })
        return formatted

# llm/templates/manager.py
import jinja2
from typing import Dict, Any, Optional, List
from dataclasses import asdict
from llm.models.base import PromptTemplate, TemplateVariable, GenerationContext

class PromptTemplateManager:
    """提示词模板管理器"""

    def __init__(self):
        self.templates: Dict[str, PromptTemplate] = {}
        self.env = jinja2.Environment(
            trim_blocks=True,
            lstrip_blocks=True
        )
        self._register_builtin_filters()

    def _register_builtin_filters(self) -> None:
        """注册内置过滤器"""
        self.env.filters['json'] = lambda x: json.dumps(x, indent=2, ensure_ascii=False)
        self.env.filters['tojson'] = lambda x: json.dumps(x)

    def register_template(self, template: PromptTemplate) -> None:
        """注册模板"""
        self.templates[template.id] = template

    def get_template(self, template_id: str) -> Optional[PromptTemplate]:
        """获取模板"""
        return self.templates.get(template_id)

    def list_templates(
        self,
        category: Optional[str] = None,
        target_platform: Optional[str] = None,
        is_system: Optional[bool] = None
    ) -> List[PromptTemplate]:
        """列出模板"""
        templates = list(self.templates.values())

        if category is not None:
            templates = [t for t in templates if t.category == category]

        if target_platform is not None:
            templates = [t for t in templates if t.target_platform == target_platform]

        if is_system is not None:
            templates = [t for t in templates if t.is_system == is_system]

        return templates

    def render(
        self,
        template_id: str,
        context: Dict[str, Any],
        validate: bool = True
    ) -> str:
        """
        渲染模板

        Args:
            template_id: 模板ID
            context: 渲染上下文
            validate: 是否验证变量

        Returns:
            str: 渲染后的内容
        """
        template = self.get_template(template_id)
        if not template:
            raise ValueError(f"Template '{template_id}' not found")

        if validate:
            self._validate_variables(template, context)

        jinja_template = self.env.from_string(template.template)
        return jinja_template.render(**context)

    def render_with_context(
        self,
        template_id: str,
        gen_context: GenerationContext,
        extra_vars: Optional[Dict[str, Any]] = None
    ) -> str:
        """
        使用生成上下文渲染模板

        Args:
            template_id: 模板ID
            gen_context: 生成上下文
            extra_vars: 额外变量

        Returns:
            str: 渲染后的内容
        """
        context = asdict(gen_context)
        if extra_vars:
            context.update(extra_vars)

        return self.render(template_id, context)

    def _validate_variables(self, template: PromptTemplate, context: Dict[str, Any]) -> None:
        """验证变量"""
        for var in template.variables:
            if var.required and var.name not in context:
                raise ValueError(f"Required variable '{var.name}' is missing")

    def create_template_from_string(
        self,
        id: str,
        name: str,
        template_str: str,
        description: Optional[str] = None,
        category: Optional[str] = None,
        variables: Optional[List[TemplateVariable]] = None
    ) -> PromptTemplate:
        """从字符串创建模板"""
        template = PromptTemplate(
            id=id,
            name=name,
            description=description,
            category=category,
            template=template_str,
            variables=variables or [],
            is_system=False
        )
        self.register_template(template)
        return template

# llm/services/llm_service.py
from typing import Dict, Any, List, Optional, AsyncGenerator
from datetime import datetime
from llm.models.base import (
    LLMProviderType,
    Message,
    GenerationResult,
    LLMConfig,
    GenerationRequest,
    GenerationContext,
    GenerationStatus
)
from llm.providers.base import LLMProvider
from llm.providers.claude import ClaudeProvider
from llm.providers.openai import OpenAIProvider
from llm.providers.mock import MockProvider
from llm.providers.gemini import GeminiProvider
from llm.providers.qwen import QwenProvider
from llm.templates.manager import PromptTemplateManager
from llm.templates.default_templates import register_default_templates

class LLMService:
    """LLM服务"""

    def __init__(self):
        self.providers: Dict[LLMProviderType, type] = {
            LLMProviderType.CLAUDE: ClaudeProvider,
            LLMProviderType.OPENAI: OpenAIProvider,
            LLMProviderType.MOCK: MockProvider,
            LLMProviderType.GEMINI: GeminiProvider,
            LLMProviderType.QWEN: QwenProvider
        }

        self.template_manager = PromptTemplateManager()
        register_default_templates(self.template_manager)

        self.active_provider: Optional[LLMProvider] = None
        self.active_config: Optional[LLMConfig] = None

    def set_active_config(self, config: LLMConfig) -> None:
        """设置活动配置"""
        provider_class = self.providers.get(config.provider)
        if not provider_class:
            raise ValueError(f"Provider '{config.provider}' not supported")

        self.active_config = config
        self.active_provider = provider_class(config)

    def get_provider(self) -> LLMProvider:
        """获取当前提供者"""
        if not self.active_provider:
            raise ValueError("No active provider configured")
        return self.active_provider

    async def generate_with_template(
        self,
        request: GenerationRequest,
        context: GenerationContext
    ) -> GenerationResult:
        """
        使用模板生成

        Args:
            request: 生成请求
            context: 生成上下文

        Returns:
            GenerationResult: 生成结果
        """
        if request.template_id:
            prompt = self.template_manager.render_with_context(
                request.template_id,
                context,
                request.variables
            )
        elif request.custom_prompt:
            prompt = request.custom_prompt
        else:
            return GenerationResult(
                success=False,
                error="Either template_id or custom_prompt is required"
            )

        result = await self.generate(
            prompt=prompt,
            max_tokens=request.max_tokens,
            temperature=request.temperature
        )

        return result

    async def generate(
        self,
        prompt: str,
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> GenerationResult:
        """
        生成内容

        Args:
            prompt: 提示词
            max_tokens: 最大Token数
            temperature: 温度参数

        Returns:
            GenerationResult: 生成结果
        """
        provider = self.get_provider()
        return await provider.generate(
            prompt=prompt,
            max_tokens=max_tokens,
            temperature=temperature,
            **kwargs
        )

    async def chat(
        self,
        messages: List[Message],
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> GenerationResult:
        """对话"""
        provider = self.get_provider()
        return await provider.chat(
            messages=messages,
            max_tokens=max_tokens,
            temperature=temperature,
            **kwargs
        )

    async def stream_chat(
        self,
        messages: List[Message],
        max_tokens: Optional[int] = None,
        temperature: Optional[float] = None,
        **kwargs
    ) -> AsyncGenerator[str, None]:
        """流式对话"""
        provider = self.get_provider()
        async for chunk in provider.stream_chat(
            messages=messages,
            max_tokens=max_tokens,
            temperature=temperature,
            **kwargs
        ):
            yield chunk

    async def generate_code(
        self,
        context: GenerationContext,
        platform: Optional[str] = None,
        template_id: Optional[str] = None
    ) -> GenerationResult:
        """
        生成代码

        Args:
            context: 生成上下文
            platform: 目标平台
            template_id: 模板ID

        Returns:
            GenerationResult: 生成结果
        """
        if not template_id:
            platform = platform or context.target_platform or "wechat_miniapp"
            template_map = {
                "wechat_miniapp": "wechat_miniapp_code",
                "h5": "h5_code",
                "react_native": "react_native_code",
                "flutter": "flutter_code"
            }
            template_id = template_map.get(platform, "wechat_miniapp_code")

        request = GenerationRequest(
            template_id=template_id,
            context=context,
            platform=platform
        )

        return await self.generate_with_template(request, context)

    async def generate_document(
        self,
        context: GenerationContext,
        template_id: str = "technical_spec"
    ) -> GenerationResult:
        """
        生成文档

        Args:
            context: 生成上下文
            template_id: 模板ID

        Returns:
            GenerationResult: 生成结果
        """
        request = GenerationRequest(template_id=template_id, context=context)
        return await self.generate_with_template(request, context)

    async def generate_claude_md(
        self,
        context: GenerationContext
    ) -> GenerationResult:
        """生成CLAUDE.md"""
        request = GenerationRequest(template_id="claude_md", context=context)
        return await self.generate_with_template(request, context)

    async def review_code(
        self,
        code: str,
        language: Optional[str] = None
    ) -> GenerationResult:
        """代码审查"""
        prompt = self.template_manager.render(
            "code_review",
            {"code": code, "language": language}
        )
        return await self.generate(prompt)

    async def explain_code(
        self,
        code: str,
        language: Optional[str] = None
    ) -> GenerationResult:
        """代码解释"""
        prompt = self.template_manager.render(
            "code_explain",
            {"code": code, "language": language}
        )
        return await self.generate(prompt)

    async def fix_bug(
        self,
        code: str,
        error_message: Optional[str] = None,
        description: Optional[str] = None
    ) -> GenerationResult:
        """Bug修复"""
        prompt = self.template_manager.render(
            "bug_fix",
            {
                "code": code,
                "error_message": error_message,
                "description": description
            }
        )
        return await self.generate(prompt)

    def get_available_providers(self) -> List[Dict[str, Any]]:
        """获取可用提供者列表"""
        return [
            {
                "type": provider_type.value,
                "name": provider_type.value.capitalize(),
                "default_model": provider.get_default_model(),
                "available_models": provider.AVAILABLE_MODELS
            }
            for provider_type, provider in self.providers.items()
        ]

    def get_available_templates(
        self,
        category: Optional[str] = None,
        target_platform: Optional[str] = None
    ) -> List[Dict[str, Any]]:
        """获取可用模板列表"""
        templates = self.template_manager.list_templates(
            category=category,
            target_platform=target_platform
        )

        return [
            {
                "id": t.id,
                "name": t.name,
                "description": t.description,
                "category": t.category,
                "target_platform": t.target_platform,
                "output_type": t.output_type.value if t.output_type else None,
                "variables": [
                    {
                        "name": v.name,
                        "type": v.type,
                        "required": v.required,
                        "description": v.description
                    }
                    for v in t.variables
                ]
            }
            for t in templates
        ]

    def count_tokens(self, text: str) -> int:
        """计算Token数"""
        try:
            return self.get_provider().count_tokens(text)
        except:
            return len(text.split())

# llm/exceptions.py
class LLMException(Exception):
    """LLM异常基类"""
    pass

class ProviderNotFoundException(LLMException):
    """提供者未找到"""
    pass

class InvalidConfigException(LLMException):
    """配置无效"""
    pass

class GenerationFailedException(LLMException):
    """生成失败"""
    pass

class TokenLimitExceededException(LLMException):
    """Token超限"""
    pass

class RateLimitExceededException(LLMException):
    """频率超限"""
    pass

class TemplateNotFoundException(LLMException):
    """模板未找到"""
    pass
```

## 前端集成

### Pinia Store

```typescript
// frontend/src/stores/llm.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import api from '@/api'

export interface LLMConfig {
  id?: number
  provider: string
  name: string
  apiKey?: string
  apiBase?: string
  model?: string
  maxTokens?: number
  temperature?: number
  isActive?: boolean
  isDefault?: boolean
}

export interface Message {
  role: 'user' | 'assistant' | 'system'
  content: string
}

export interface GenerationResult {
  success: boolean
  content?: string
  error?: string
  promptTokens?: number
  completionTokens?: number
  totalTokens?: number
  model?: string
  durationMs?: number
}

export const useLLMStore = defineStore('llm', () => {
  const configs = ref<LLMConfig[]>([])
  const activeConfig = ref<LLMConfig | null>(null)
  const isGenerating = ref(false)
  const generationHistory = ref<GenerationResult[]>([])

  const providers = computed(() => [
    { type: 'claude', name: 'Claude' },
    { type: 'openai', name: 'OpenAI' },
    { type: 'gemini', name: 'Gemini' },
    { type: 'qwen', name: '通义千问' },
    { type: 'mock', name: 'Mock (测试)' }
  ])

  async function fetchConfigs() {
    const response = await api.get('/llm/configs')
    configs.value = response.data
  }

  async function createConfig(config: LLMConfig) {
    await api.post('/llm/configs', config)
    await fetchConfigs()
  }

  async function updateConfig(id: number, config: Partial<LLMConfig>) {
    await api.put(`/llm/configs/${id}`, config)
    await fetchConfigs()
  }

  async function activateConfig(id: number) {
    await api.post(`/llm/configs/${id}/activate`)
    await fetchConfigs()
  }

  async function generateCode(params: any): Promise<GenerationResult> {
    isGenerating.value = true
    try {
      const response = await api.post('/llm/generate/code', params)
      generationHistory.value.unshift(response.data)
      return response.data
    } finally {
      isGenerating.value = false
    }
  }

  async function generateDocument(params: any): Promise<GenerationResult> {
    isGenerating.value = true
    try {
      const response = await api.post('/llm/generate/document', params)
      generationHistory.value.unshift(response.data)
      return response.data
    } finally {
      isGenerating.value = false
    }
  }

  async function generateClaudeMd(params: any): Promise<GenerationResult> {
    isGenerating.value = true
    try {
      const response = await api.post('/llm/generate/claude-md', params)
      generationHistory.value.unshift(response.data)
      return response.data
    } finally {
      isGenerating.value = false
    }
  }

  async function chat(messages: Message[]): Promise<GenerationResult> {
    isGenerating.value = true
    try {
      const response = await api.post('/llm/chat', { messages })
      generationHistory.value.unshift(response.data)
      return response.data
    } finally {
      isGenerating.value = false
    }
  }

  async function reviewCode(code: string, language?: string): Promise<GenerationResult> {
    isGenerating.value = true
    try {
      const response = await api.post('/llm/review/code', { code, language })
      generationHistory.value.unshift(response.data)
      return response.data
    } finally {
      isGenerating.value = false
    }
  }

  async function explainCode(code: string, language?: string): Promise<GenerationResult> {
    isGenerating.value = true
    try {
      const response = await api.post('/llm/explain/code', { code, language })
      generationHistory.value.unshift(response.data)
      return response.data
    } finally {
      isGenerating.value = false
    }
  }

  async function fixBug(code: string, errorMessage?: string, description?: string): Promise<GenerationResult> {
    isGenerating.value = true
    try {
      const response = await api.post('/llm/fix/bug', { code, errorMessage, description })
      generationHistory.value.unshift(response.data)
      return response.data
    } finally {
      isGenerating.value = false
    }
  }

  return {
    configs,
    activeConfig,
    isGenerating,
    generationHistory,
    providers,
    fetchConfigs,
    createConfig,
    updateConfig,
    activateConfig,
    generateCode,
    generateDocument,
    generateClaudeMd,
    chat,
    reviewCode,
    explainCode,
    fixBug
  }
})
```

## 安全考虑

### 1. API Key加密存储
- 使用AES加密存储API Key
- 加密密钥从环境变量获取
- 不在日志中记录API Key

### 2. 访问控制
- 配置只属于创建用户
- 生成操作需要认证
- 生成日志记录用户行为

### 3. 限流控制
- API调用频率限制
- Token使用量限制
- 防止滥用

### 4. 内容审核
- 敏感内容过滤
- 生成内容审计
