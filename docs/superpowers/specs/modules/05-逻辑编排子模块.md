# 子模块 05: 逻辑编排子模块设计文档

## 概述

负责事件-动作绑定、数据流配置。采用简单的"当发生X时，执行Y"模式，提供可视化的逻辑编排能力。支持条件判断、循环、错误处理等复杂逻辑，让用户无需编写代码即可实现应用的交互逻辑。

### 核心能力
- **事件绑定**: 支持组件事件、页面生命周期事件、自定义事件
- **动作系统**: 10+内置动作类型，支持自定义扩展
- **表达式引擎**: 数据绑定、变量引用、路径解析
- **条件逻辑**: 简单/复合条件判断（AND/OR）
- **流程控制**: 条件分支、循环执行
- **错误处理**: try-catch、成功/失败回调

## 技术栈

- **前端**: TypeScript、Vue 3
- **后端**: Python、Pydantic、FastAPI
- **表达式引擎**: 自定义解析器（安全、可控）
- **数据验证**: JSON Schema
- **类型安全**: TypeScript 完整类型定义

```python
class ConditionBase(BaseModel):
    type: ConditionType


class SimpleCondition(ConditionBase):
    type: ConditionType = ConditionType.SIMPLE
    left: str
    operator: ConditionOperator
    right: Any

    type: ConditionType
    children: List["ConditionType"]
```

```typescript
// frontend/src/modules/logic/utils/expression-evaluator.ts
export class ExpressionEvaluator {
  context: ExpressionContext
  private cache: Map<string, any> = new Map()

  constructor(context: ExpressionContext) {
    this.context = context
  }

  evaluate(expression: string): any {
    if (!expression || typeof expression !== 'string') {
      return expression
    }

    // 完整表达式模式：{{path}}
    const fullMatch = expression.match(/^\{\{([^}]+)\}\}$/)
    if (fullMatch) {
      return this.resolvePath(fullMatch[1].trim())
    }

    // 模板字符串模式：Hello {{name}}!
    const regex = /\{\{([^}]+)\}\}/g
    let result = expression
    let match

    while ((match = regex.exec(expression)) !== null) {
      const value = this.resolvePath(match[1].trim())
      const stringValue = this.valueToString(value)
      result = result.replace(match[0], stringValue)
    }

    return result
  }

  evaluateObject<T = Record<string, any>>(obj: T): T {
    if (obj === null || obj === undefined) return obj
    if (typeof obj !== 'object') return obj

    if (Array.isArray(obj)) {
      return obj.map(item => this.evaluateObject(item)) as unknown as T
    }

    const result: Record<string, any> = {}
    for (const [key, value] of Object.entries(obj)) {
      if (typeof value === 'string') {
        result[key] = this.evaluate(value)
      } else if (typeof value === 'object') {
        result[key] = this.evaluateObject(value)
      } else {
        result[key] = value
      }
    }
    return result as T
  }

  evaluateCondition(condition: Condition | undefined): boolean {
    if (!condition) return true

    switch (condition.type) {
      case 'and':
        return condition.children?.every(c => this.evaluateCondition(c)) ?? true
      case 'or':
        return condition.children?.some(c => this.evaluateCondition(c)) ?? false
      case 'not':
        return !this.evaluateCondition(condition.children?.[0])
      case 'simple':
        return this.evaluateSimpleCondition(condition)
      default:
        return true
    }
  }

  private evaluateSimpleCondition(condition: Condition): boolean {
    const leftVal = this.evaluate(condition.left ?? '')
    const rightVal = this.evaluate(condition.right ?? '')

    switch (condition.operator) {
      case '==':
        return this.isEqual(leftVal, rightVal)
      case '!=':
        return !this.isEqual(leftVal, rightVal)
      case '>':
        return leftVal > rightVal
      case '<':
        return leftVal < rightVal
      case '>=':
        return leftVal >= rightVal
      case '<=':
        return leftVal <= rightVal
      case 'includes':
        return this.includes(leftVal, rightVal)
      case 'excludes':
        return !this.includes(leftVal, rightVal)
      case 'startsWith':
        return String(leftVal).startsWith(String(rightVal))
      case 'endsWith':
        return String(leftVal).endsWith(String(rightVal))
      case 'matches':
        try {
          const regex = new RegExp(String(rightVal))
          return regex.test(String(leftVal))
        } catch {
          return false
        }
      case 'exists':
        return leftVal !== null && leftVal !== undefined
      default:
        return false
    }
  }

  private resolvePath(path: string): any {
    // 检查是否是函数调用
    const funcMatch = path.match(/^(\w+)\((.*)\)$/)
    if (funcMatch) {
      const funcName = funcMatch[1]
      const args = this.parseFunctionArgs(funcMatch[2])
      return this.callFunction(funcName, args)
    }

    const parts = this.splitPath(path)
    let current: any = null

    // 解析第一部分确定源
    const firstPart = parts[0]
    if (firstPart === 'storage') {
      const storageType = parts[1] as 'local' | 'session' | 'app'
      current = this.context.storage[storageType] || {}
      parts.splice(0, 2)
    } else if (firstPart === 'routeParams') {
      current = this.context.routeParams
      parts.shift()
    } else if (firstPart === 'loopItems') {
      current = this.context.loopItems
      parts.shift()
    } else if (firstPart === 'eventData') {
      current = this.context.eventData
      parts.shift()
    } else if (firstPart === 'user') {
      current = this.context.user
      parts.shift()
    } else if (firstPart === 'globalState') {
      current = this.context.globalStates
      parts.shift()
    } else if (firstPart === 'pageState') {
      current = this.context.pageStates
      parts.shift()
    } else {
      if (this.context.components[firstPart] !== undefined) {
        current = this.context.components[firstPart]
        parts.shift()
      } else {
        if (this.context.pageStates[firstPart] !== undefined) {
          current = this.context.pageStates[firstPart]
          parts.shift()
        } else if (this.context.apiResponses[firstPart] !== undefined) {
          current = this.context.apiResponses[firstPart]
          parts.shift()
        } else {
          return undefined
        }
      }
    }

    for (const part of parts) {
      if (current && typeof current === 'object') {
        const arrayMatch = part.match(/^(\d+)$/)
        if (arrayMatch) {
          const index = parseInt(arrayMatch[1], 10)
          if (Array.isArray(current)) {
            current = current[index]
            continue
          }
        }
        if (part === '*' && Array.isArray(current)) {
          if (parts.length > parts.indexOf(part) + 1) {
            const remainingParts = parts.slice(parts.indexOf(part) + 1)
            return current.map(item => {
              let itemValue = item
              for (const p of remainingParts) {
                if (itemValue && typeof itemValue === 'object') {
                  itemValue = itemValue[p]
                } else {
                  return undefined
                }
              }
              return itemValue
            })
          }
          return current
        }
        current = current[part]
      } else {
        return undefined
      }
    }

    return current
  }

  private splitPath(path: string): string[] {
    const result: string[] = []
    let current = ''
    let inQuotes = false
    let quoteChar = ''

    for (const char of path) {
      if ((char === "'" || char === '"') && !inQuotes) {
        inQuotes = true
        quoteChar = char
      } else if (char === quoteChar && inQuotes) {
        inQuotes = false
        quoteChar = ''
      } else if (char === '.' && !inQuotes) {
        result.push(current)
        current = ''
      } else {
        current += char
      }
    }
    if (current) {
      result.push(current)
    }
    return result
  }

  private parseFunctionArgs(argsStr: string): any[] {
    if (!argsStr.trim()) return []

    const args: any[] = []
    let current = ''
    let depth = 0
    let inQuotes = false
    let quoteChar = ''

    for (const char of argsStr) {
      if ((char === "'" || char === '"') && !inQuotes) {
        inQuotes = true
        quoteChar = char
        current += char
      } else if (char === quoteChar && inQuotes) {
        inQuotes = false
        quoteChar = ''
        current += char
      } else if (char === '(' && !inQuotes) {
        depth++
        current += char
      } else if (char === ')' && !inQuotes) {
        depth--
        current += char
      } else if (char === ',' && !inQuotes && depth === 0) {
        args.push(this.parseArgValue(current.trim()))
        current = ''
      } else {
        current += char
      }
    }
    if (current.trim()) {
      args.push(this.parseArgValue(current.trim()))
    }
    return args
  }

  private parseArgValue(value: string): any {
    if ((value.startsWith("'") && value.endsWith("'")) ||
        (value.startsWith('"') && value.endsWith('"'))) {
      return value.slice(1, -1)
    }
    if (/^-?\d+(\.\d+)?$/.test(value)) {
      return parseFloat(value)
    }
    if (value === 'true') return true
    if (value === 'false') return false
    if (value === 'null') return null
    if (value === 'undefined') return undefined
    return this.resolvePath(value)
  }

  private callFunction(funcName: string, args: any[]): any {
    const functions: Record<string, (...args: any[]) => any> = {
      length: (x: any) => {
        if (Array.isArray(x) || typeof x === 'string') {
          return x.length
        }
        if (x && typeof x === 'object') {
          return Object.keys(x).length
        }
        return 0
      },
      isEmpty: (x: any) => {
        if (x === null || x === undefined) return true
        if (x === '') return true
        if (Array.isArray(x) && x.length === 0) return true
        if (typeof x === 'object' && Object.keys(x).length === 0) return true
        return false
      },
      join: (arr: any[], sep: string = ',') => {
        if (Array.isArray(arr)) {
          return arr.join(sep)
        }
        return ''
      },
      split: (str: string, sep: string = ',') => {
        if (typeof str === 'string') {
          return str.split(sep)
        }
        return []
      },
      toLowerCase: (str: string) => String(str).toLowerCase(),
      toUpperCase: (str: string) => String(str).toUpperCase(),
      trim: (str: string) => String(str).trim(),
      default: (val: any, fallback: any) => val !== undefined && val !== null ? val : fallback,
      now: () => new Date().toISOString(),
      formatDate: (date: string | Date, fmt: string = 'YYYY-MM-DD') => {
        const d = typeof date === 'string' ? new Date(date) : date
        if (isNaN(d.getTime())) return ''
        const year = d.getFullYear()
        const month = String(d.getMonth() + 1).padStart(2, '0')
        const day = String(d.getDate()).padStart(2, '0')
        const hour = String(d.getHours()).padStart(2, '0')
        const minute = String(d.getMinutes()).padStart(2, '0')
        const second = String(d.getSeconds()).padStart(2, '0')
        return fmt
          .replace('YYYY', String(year))
          .replace('MM', month)
          .replace('DD', day)
          .replace('HH', hour)
          .replace('mm', minute)
          .replace('ss', second)
      },
      number: (val: any) => Number(val),
      string: (val: any) => String(val),
      boolean: (val: any) => Boolean(val)
    }

    const func = functions[funcName]
    if (func) {
      return func(...args)
    }
    console.warn(`Unknown function: ${funcName}`)
    return undefined
  }

  private isEqual(a: any, b: any): boolean {
    if (a === b) return true
    if (a != null && b != null) {
      if (typeof a === 'number' || typeof b === 'number') {
        return Number(a) === Number(b)
      }
    }
    return String(a) === String(b)
  }

  private includes(container: any, item: any): boolean {
    if (Array.isArray(container)) {
      return container.includes(item)
    }
    if (typeof container === 'string') {
      return container.includes(String(item))
    }
    if (container && typeof container === 'object') {
      return String(item) in container
    }
    return false
  }

  private valueToString(value: any): string {
    if (value === null || value === undefined) return ''
    if (typeof value === 'object') {
      try {
        return JSON.stringify(value)
      } catch {
        return String(value)
      }
    }
    return String(value)
  }

  clone(): ExpressionEvaluator {
    return new ExpressionEvaluator({ ...this.context })
  }

  extend(newContext: Partial<ExpressionContext>): ExpressionEvaluator {
    return new ExpressionEvaluator({
      ...this.context,
      ...newContext
    })
  }
}
```

```typescript
export interface EventBus {

export class ActionExecutor {
  private deps: ActionExecutorDeps
  private state: ExecutionState

  constructor(deps: ActionExecutorDeps) {
    this.deps = deps
    this.state = {
      stopped: false,
      currentActionIndex: 0,
      errors: [],
      results: {}
    }
  }

  async executeActions(actions: Action[]): Promise<ExecutionResult> {
    this.state = {
      stopped: false,
      currentActionIndex: 0,
      errors: [],
      results: {}
    }

    for (let i = 0; i < actions.length; i++) {
      if (this.state.stopped) break

      const action = actions[i]
      this.state.currentActionIndex = i

      try {
        await this.executeAction(action)
      } catch (error) {
        this.state.errors.push(error as Error)

        const errorHandling = action.errorHandling
        if (!errorHandling || errorHandling.strategy === 'stop') {
          throw error
        }
        if (errorHandling.strategy === 'custom' && errorHandling.errorActions) {
          await this.executeActions(errorHandling.errorActions)
        }
        if (errorHandling.strategy === 'continue') {
          continue
        }
      }
    }

    return {
      success: this.state.errors.length === 0,
      errors: this.state.errors,
      results: this.state.results
    }
  }

  stop(): void {
    this.state.stopped = true
  }

  private async executeAction(action: Action): Promise<void> {
    if (action.condition) {
      const conditionMet = this.deps.evaluator.evaluateCondition(action.condition)
      if (!conditionMet) {
        return
      }
    }

    switch (action.type) {
      case 'call_api':
        await this.executeCallApi(action.config as CallApiActionConfig)
        break
      case 'set_data':
        this.executeSetData(action.config as SetDataActionConfig)
        break
      case 'set_state':
        this.executeSetState(action.config as SetStateActionConfig)
        break
      case 'navigate':
        this.executeNavigate(action.config as NavigateActionConfig)
        break
      case 'show_message':
        this.executeShowMessage(action.config as ShowMessageActionConfig)
        break
      case 'show_loading':
        this.executeShowLoading()
        break
      case 'hide_loading':
        this.executeHideLoading()
        break
      case 'set_storage':
        this.executeSetStorage(action.config as SetStorageActionConfig)
        break
      case 'get_storage':
        this.executeGetStorage(action.config as GetStorageActionConfig)
        break
      case 'remove_storage':
        this.executeRemoveStorage(action.config as RemoveStorageActionConfig)
        break
      case 'condition':
        await this.executeCondition(action.config as ConditionActionConfig)
        break
      case 'loop':
        await this.executeLoop(action.config as LoopActionConfig)
        break
      case 'delay':
        await this.executeDelay(action.config as DelayActionConfig)
        break
      case 'console_log':
        this.executeConsoleLog(action.config as ConsoleLogActionConfig)
        break
      case 'set_component_prop':
        this.executeSetComponentProp(action.config as SetComponentPropActionConfig)
        break
      case 'trigger_event':
        this.executeTriggerEvent(action.config as TriggerEventActionConfig)
        break
      case 'confirm':
        await this.executeConfirm(action.config as ConfirmActionConfig)
        break
    }
  }

  private async executeCallApi(config: CallApiActionConfig): Promise<void> {
    const params = this.deps.evaluator.evaluateObject(config.params)
    const body = config.body ? this.deps.evaluator.evaluateObject(config.body) : undefined
    const headers = config.headers ? this.deps.evaluator.evaluateObject(config.headers) : undefined

    try {
      this.deps.messageService.showLoading('加载中...')

      const response = await this.deps.apiCaller.call({
        apiId: config.apiId,
        params,
        body,
        headers,
        method: config.method,
        timeout: config.timeout
      })

      this.deps.messageService.hideLoading()

      if (config.assignTo) {
        this.deps.stateManager.set(config.assignTo, response.data)
      }

      if (config.onSuccess) {
        if (config.onSuccess.showSuccessMessage) {
          const message = this.deps.evaluator.evaluate(
            config.onSuccess.successMessage || '操作成功'
          )
          this.deps.messageService.success(message)
        }
        if (config.onSuccess.actions && config.onSuccess.actions.length > 0) {
          const successEvaluator = this.deps.evaluator.extend({
            apiResponses: {
              ...this.deps.evaluator.context.apiResponses,
              [config.apiId]: response
            }
          })
          const successExecutor = new ActionExecutor({
            ...this.deps,
            evaluator: successEvaluator
          })
          await successExecutor.executeActions(config.onSuccess.actions)
        }
      }
    } catch (error) {
      this.deps.messageService.hideLoading()

      if (config.onError) {
        if (config.onError.showErrorMessage) {
          const message = config.onError.useApiError
            ? (error as ApiError).message
            : this.deps.evaluator.evaluate(config.onError.errorMessage || '操作失败')
          this.deps.messageService.error(message)
        }
        if (config.onError.actions && config.onError.actions.length > 0) {
          await this.executeActions(config.onError.actions)
        }
      } else {
        throw error
      }
    }
  }

  private executeSetData(config: SetDataActionConfig): void {
    const value = this.deps.evaluator.evaluate(config.value)
    const target = this.deps.evaluator.evaluate(config.target)

    switch (config.mode) {
      case 'set':
        this.deps.stateManager.set(config.target, value)
        break
      case 'append':
        this.deps.stateManager.append(config.target, value)
        break
      case 'prepend':
        this.deps.stateManager.prepend(config.target, value)
        break
      case 'delete':
        this.deps.stateManager.delete(config.target)
        break
      case 'toggle':
        this.deps.stateManager.toggle(config.target)
        break
    }
  }

  private executeSetState(config: SetStateActionConfig): void {
    const value = this.deps.evaluator.evaluate(config.value)
    this.deps.stateManager.setState(config.stateId, value)
  }

  private executeNavigate(config: NavigateActionConfig): void {
    const params = this.deps.evaluator.evaluateObject(config.params)
    this.deps.navigator.navigate({
      pageId: config.pageId,
      pagePath: config.pagePath,
      params,
      type: config.type,
      delta: config.delta
    })
  }

  private executeShowMessage(config: ShowMessageActionConfig): void {
    const content = this.deps.evaluator.evaluate(config.content)
    this.deps.messageService.show({
      content,
      type: config.type,
      duration: config.duration,
      icon: config.icon,
      position: config.position
    })
  }

  private executeShowLoading(): void {
    this.deps.messageService.showLoading()
  }

  private executeHideLoading(): void {
    this.deps.messageService.hideLoading()
  }

  private executeSetStorage(config: SetStorageActionConfig): void {
    const value = this.deps.evaluator.evaluate(config.value)
    this.deps.storage.set(config.key, value, config.storageType, config.expires)
  }

  private executeGetStorage(config: GetStorageActionConfig): void {
    const value = this.deps.storage.get(config.key, config.storageType)
    if (config.target) {
      this.deps.stateManager.set(config.target, value)
    }
  }

  private executeRemoveStorage(config: RemoveStorageActionConfig): void {
    this.deps.storage.remove(config.key, config.storageType)
  }

  private async executeCondition(config: ConditionActionConfig): Promise<void> {
    const result = this.deps.evaluator.evaluateCondition(config.condition)
    if (result) {
      await this.executeActions(config.trueActions)
    } else if (config.falseActions) {
      await this.executeActions(config.falseActions)
    }
  }

  private async executeLoop(config: LoopActionConfig): Promise<void> {
    const source = this.deps.evaluator.evaluate(config.source)

    if (!Array.isArray(source)) {
      console.warn('Loop source is not an array:', source)
      return
    }

    const maxIterations = config.maxIterations ?? 1000
    const iterations = Math.min(source.length, maxIterations)

    for (let index = 0; index < iterations; index++) {
      if (this.state.stopped) break

      const item = source[index]
      const loopContext: Partial<ExpressionContext> = {
        loopItems: {
          [config.itemName]: item,
          ...(config.indexName ? { [config.indexName]: index } : {})
        }
      }

      const loopEvaluator = this.deps.evaluator.extend(loopContext)
      const loopExecutor = new ActionExecutor({
        ...this.deps,
        evaluator: loopEvaluator
      })

      if (config.breakCondition) {
        const shouldBreak = loopEvaluator.evaluateCondition(config.breakCondition)
        if (shouldBreak) break
      }

      await loopExecutor.executeActions(config.actions)
    }
  }

  private executeDelay(config: DelayActionConfig): Promise<void> {
    return new Promise(resolve => {
      setTimeout(() => {
        if (!this.state.stopped) {
          resolve()
        }
      }, config.duration)
    })
  }

  private executeConsoleLog(config: ConsoleLogActionConfig): void {
    const message = this.deps.evaluator.evaluate(config.message)
    const values = config.values?.map(v => this.deps.evaluator.evaluate(v)) || []

    const consoleFunc = console[config.level] || console.log
    consoleFunc(message, ...values)
  }

  private executeSetComponentProp(config: SetComponentPropActionConfig): void {
    const value = this.deps.evaluator.evaluate(config.value)
    this.deps.componentManager.setProp(config.componentId, config.propName, value)
  }

  private executeTriggerEvent(config: TriggerEventActionConfig): void {
    const eventData = config.eventData
      ? this.deps.evaluator.evaluateObject(config.eventData)
      : undefined
    this.deps.eventBus.trigger(config.eventName, eventData, config.target)
  }

  private executeConfirm(config: ConfirmActionConfig): Promise<void> {
    return new Promise((resolve) => {
      this.deps.messageService.confirm({
        title: config.title,
        content: this.deps.evaluator.evaluate(config.content),
        confirmText: config.confirmText,
        cancelText: config.cancelText,
        onConfirm: async () => {
          await this.executeActions(config.confirmActions)
          resolve()
        },
        onCancel: async () => {
          if (config.cancelActions) {
            await this.executeActions(config.cancelActions)
          }
          resolve()
        }
      })
    })
  }
}
```

```python
# backend/src/modules/logic/services/validation_service.py
from typing import List, Dict, Any, Optional
from backend.src.modules.logic.schemas.event_binding import (
    Action, ActionType, Condition, ConditionType, ValidationResult,
    ValidationError, ValidationWarning
)


class ValidationService:
    """逻辑编排验证服务"""

    def __init__(self, db):
        self.db = db
        self.action_validators = {
            ActionType.CALL_API: self._validate_call_api,
            ActionType.SET_DATA: self._validate_set_data,
            ActionType.NAVIGATE: self._validate_navigate,
            ActionType.SHOW_MESSAGE: self._validate_show_message,
            ActionType.CONDITION: self._validate_condition_action,
            ActionType.LOOP: self._validate_loop_action,
        }

    def validate_actions(self, actions: List[Action]) -> ValidationResult:
        """验证动作列表"""
        errors: List[ValidationError] = []
        warnings: List[ValidationWarning] = []

        for i, action in enumerate(actions):
            path = f"actions.{i}"

            if action.condition:
                cond_errors, cond_warnings = self._validate_condition(
                    action.condition, f"{path}.condition"
                )
                errors.extend(cond_errors)
                warnings.extend(cond_warnings)

            validator = self.action_validators.get(action.type)
            if validator:
                act_errors, act_warnings = validator(action.config, path)
                errors.extend(act_errors)
                warnings.extend(act_warnings)
            else:
                warnings.append(ValidationWarning(
                    path=path,
                    message=f"Unknown action type: {action.type}"
                ))

            if action.type == ActionType.CONDITION:
                if "trueActions" in action.config:
                    true_result = self.validate_actions(action.config["trueActions"])
                    errors.extend([
                        ValidationError(path=f"{path}.trueActions.{e.path}", message=e.message)
                        for e in true_result.errors
                    ])
                if "falseActions" in action.config:
                    false_result = self.validate_actions(action.config["falseActions"])
                    errors.extend([
                        ValidationError(path=f"{path}.falseActions.{e.path}", message=e.message)
                        for e in false_result.errors
                    ])

            if action.type == ActionType.LOOP:
                if "actions" in action.config:
                    loop_result = self.validate_actions(action.config["actions"])
                    errors.extend([
                        ValidationError(path=f"{path}.actions.{e.path}", message=e.message)
                        for e in loop_result.errors
                    ])

            if action.error_handling and action.error_handling.error_actions:
                error_actions_result = self.validate_actions(action.error_handling.error_actions)
                errors.extend([
                    ValidationError(path=f"{path}.errorHandling.errorActions.{e.path}", message=e.message)
                    for e in error_actions_result.errors
                ])

        return ValidationResult(
            valid=len(errors) == 0,
            errors=errors,
            warnings=warnings
        )

    def _validate_condition(self, condition: Condition, path: str):
        errors = []
        warnings = []

        if condition.type in [ConditionType.AND, ConditionType.OR]:
            if not condition.children:
                warnings.append(ValidationWarning(
                    path=path,
                    message=f"{condition.type.value} 条件没有子条件"
                ))
            for i, child in enumerate(condition.children or []):
                child_errors, child_warnings = self._validate_condition(
                    child, f"{path}.children.{i}"
                )
                errors.extend(child_errors)
                warnings.extend(child_warnings)
        elif condition.type == ConditionType.SIMPLE:
            if not condition.left or not condition.operator:
                errors.append(ValidationError(
                    path=path,
                    message="简单条件缺少 left/operator"
                ))
            if condition.operator not in ["exists"] and condition.right is None:
                errors.append(ValidationError(
                    path=path,
                    message="简单条件缺少 right"
                ))

        return errors, warnings

    def _validate_call_api(self, config: Dict[str, Any], path: str):
        errors = []
        warnings = []

        if "apiId" not in config or not config["apiId"]:
            errors.append(ValidationError(
                path=f"{path}.apiId",
                message="缺少 apiId"
            ))

        return errors, warnings

    def _validate_set_data(self, config: Dict[str, Any], path: str):
        errors = []
        warnings = []

        if "target" not in config or not config["target"]:
            errors.append(ValidationError(
                path=f"{path}.target",
                message="缺少 target"
            ))
        if "value" not in config:
            errors.append(ValidationError(
                path=f"{path}.value",
                message="缺少 value"
            ))
        if "mode" in config and config["mode"] not in ["set", "append", "prepend", "delete", "toggle"]:
            errors.append(ValidationError(
                path=f"{path}.mode",
                message="无效的 mode 值"
            ))

        return errors, warnings

    def _validate_navigate(self, config: Dict[str, Any], path: str):
        errors = []
        warnings = []

        nav_type = config.get("type", "push")
        if nav_type == "back":
            pass
        elif "pageId" not in config and "pagePath" not in config:
            errors.append(ValidationError(
                path=path,
                message="需要 pageId 或 pagePath"
            ))

        return errors, warnings

    def _validate_show_message(self, config: Dict[str, Any], path: str):
        errors = []
        warnings = []

        if "content" not in config:
            errors.append(ValidationError(
                path=f"{path}.content",
                message="缺少 content"
            ))

        return errors, warnings

    def _validate_condition_action(self, config: Dict[str, Any], path: str):
        errors = []
        warnings = []

        if "condition" not in config:
            errors.append(ValidationError(
                path=f"{path}.condition",
                message="缺少 condition"
            ))
        if "trueActions" not in config:
            errors.append(ValidationError(
                path=f"{path}.trueActions",
                message="缺少 trueActions"
            ))

        return errors, warnings

    def _validate_loop_action(self, config: Dict[str, Any], path: str):
        errors = []
        warnings = []

        if "source" not in config:
            errors.append(ValidationError(
                path=f"{path}.source",
                message="缺少 source"
            ))
        if "itemName" not in config:
            errors.append(ValidationError(
                path=f"{path}.itemName",
                message="缺少 itemName"
            ))
        if "actions" not in config:
            errors.append(ValidationError(
                path=f"{path}.actions",
                message="缺少 actions"
            ))

        return errors, warnings
```

```python
class EventBindingService:
    """事件绑定服务"""

    def __init__(self, db: Session):

        self.validation_service = ValidationService(db)

    def get_by_project(
        self, project_id: int, page_id: Optional[int] = None,
        component_id: Optional[str] = None, enabled_only: bool = False
    ) -> List[EventBindingResponse]:
        """获取项目的事件绑定"""
        query = self.db.query(EventBindingDB).filter(EventBindingDB.project_id == project_id)

        if page_id is not None:
            query = query.filter(EventBindingDB.page_id == page_id)
        if component_id is not None:
            query = query.filter(EventBindingDB.component_id == component_id)
        if enabled_only:
            query = query.filter(EventBindingDB.enabled == True)

        bindings = query.order_by(EventBindingDB.sort_order).all()
        return [EventBindingResponse.model_validate(b) for b in bindings]

    def get_by_id(self, binding_id: int, project_id: int) -> Optional[EventBindingResponse]:
        """根据ID获取事件绑定"""
        binding = self.db.query(EventBindingDB).filter(
            and_(EventBindingDB.id == binding_id, EventBindingDB.project_id == project_id)
        ).first()
        return EventBindingResponse.model_validate(binding) if binding else None

    def create(self, project_id: int, data: EventBindingCreate) -> EventBindingResponse:
        """创建事件绑定"""
        validation_result = self.validation_service.validate_actions(data.actions)
        if not validation_result.valid:
            raise ValueError(f"Validation failed: {validation_result.errors}")

        binding = EventBindingDB(
            project_id=project_id,
            component_id=data.component_id,
            page_id=data.page_id,
            event_name=data.event_name,
            event_source=data.event_source.value,
            condition=data.condition.model_dump() if data.condition else None,
            enabled=data.enabled,
            sort_order=data.sort_order,
            description=data.description,
            action_config=[a.model_dump() for a in data.actions]
        )
        self.db.add(binding)
        self.db.commit()
        self.db.refresh(binding)
        return EventBindingResponse.model_validate(binding)

    def update(self, binding_id: int, project_id: int, data: EventBindingUpdate) -> Optional[EventBindingResponse]:
        """更新事件绑定"""
        binding = self.db.query(EventBindingDB).filter(
            and_(EventBindingDB.id == binding_id, EventBindingDB.project_id == project_id)
        ).first()
        if not binding:
            return None

        if data.actions:
            validation_result = self.validation_service.validate_actions(data.actions)
            if not validation_result.valid:
                raise ValueError(f"Validation failed: {validation_result.errors}")

        update_data = data.model_dump(exclude_unset=True)
        if "actions" in update_data:
            update_data["action_config"] = [a.model_dump() for a in update_data.pop("actions")]
        if "condition" in update_data and update_data["condition"]:
            update_data["condition"] = update_data["condition"].model_dump()

        for key, value in update_data.items():
            setattr(binding, key, value)

        binding.updated_at = datetime.utcnow()
        self.db.commit()
        self.db.refresh(binding)
        return EventBindingResponse.model_validate(binding)

    def toggle_enabled(self, binding_id: int, project_id: int, enabled: bool) -> Optional[EventBindingResponse]:
        """切换启用状态"""
        binding = self.db.query(EventBindingDB).filter(
            and_(EventBindingDB.id == binding_id, EventBindingDB.project_id == project_id)
        ).first()
        if not binding:
            return None

        binding.enabled = enabled
        binding.updated_at = datetime.utcnow()
        self.db.commit()
        self.db.refresh(binding)
        return EventBindingResponse.model_validate(binding)

    def delete(self, binding_id: int, project_id: int) -> bool:
        """删除事件绑定"""
        binding = self.db.query(EventBindingDB).filter(
            and_(EventBindingDB.id == binding_id, EventBindingDB.project_id == project_id)
        ).first()
        if not binding:
            return False

        self.db.delete(binding)
        self.db.commit()
        return True

    def duplicate(self, binding_id: int, project_id: int, new_name: Optional[str] = None) -> Optional[EventBindingResponse]:
        """复制事件绑定"""
        binding = self.db.query(EventBindingDB).filter(
            and_(EventBindingDB.id == binding_id, EventBindingDB.project_id == project_id)
        ).first()
        if not binding:
            return None

        new_binding = EventBindingDB(
            project_id=project_id,
            component_id=binding.component_id,
            page_id=binding.page_id,
            event_name=binding.event_name,
            event_source=binding.event_source,
            condition=binding.condition,
            enabled=False,
            sort_order=binding.sort_order,
            description=new_name or f"{binding.description or 'Copy'} (copy)",
            action_config=binding.action_config
        )
        self.db.add(new_binding)
        self.db.commit()
        self.db.refresh(new_binding)
        return EventBindingResponse.model_validate(new_binding)

    def validate_config(self, actions: List[Action], condition: Optional[Condition] = None) -> ValidationResult:
        """验证配置"""
        action_result = self.validation_service.validate_actions(actions)

        if condition:
            cond_errors, cond_warnings = self.validation_service._validate_condition(condition, "condition")
            action_result.errors.extend(cond_errors)
            action_result.warnings.extend(cond_warnings)
            action_result.valid = len(action_result.errors) == 0
```

```typescript
// frontend/src/modules/logic/stores/logic.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import api from '@/api'
import type {
  EventBinding, LogicVariable, DataTransformer, ExecutionLog,
  Action, Condition, ActionType, EventSource
} from '../types'

export const useLogicStore = defineStore('logic', () => {
  const eventBindings = ref<EventBinding[]>([])
  const variables = ref<LogicVariable[]>([])
  const transformers = ref<DataTransformer[]>([])
  const executionLogs = ref<ExecutionLog[]>([])
  const currentBinding = ref<EventBinding | null>(null)
  const loading = ref(false)

  const enabledBindings = computed(() =>
    eventBindings.value.filter(b => b.enabled)
  )

  async function fetchEventBindings(
    projectId: number,
    options?: { pageId?: number; componentId?: string; enabledOnly?: boolean }
  ) {
    loading.value = true
    try {
      const params: any = {}
      if (options?.pageId) params.pageId = options.pageId
      if (options?.componentId) params.componentId = options.componentId
      if (options?.enabledOnly) params.enabledOnly = options.enabledOnly

      const response = await api.get(`/logic/projects/${projectId}/event-bindings`, { params })
      eventBindings.value = response.data
    } finally {
      loading.value = false
    }
  }

  async function getEventBinding(projectId: number, bindingId: number) {
    const response = await api.get(`/logic/projects/${projectId}/event-bindings/${bindingId}`)
    currentBinding.value = response.data
    return response.data
  }

  async function createEventBinding(projectId: number, data: Omit<EventBinding, 'id' | 'createdAt' | 'updatedAt'>) {
    const response = await api.post(`/logic/projects/${projectId}/event-bindings`, data)
    eventBindings.value.push(response.data)
    return response.data
  }

  async function updateEventBinding(projectId: number, bindingId: number, data: Partial<EventBinding>) {
    const response = await api.put(`/logic/projects/${projectId}/event-bindings/${bindingId}`, data)
    const index = eventBindings.value.findIndex(b => b.id === bindingId)
    if (index !== -1) {
      eventBindings.value[index] = response.data
    }
    if (currentBinding.value?.id === bindingId) {
      currentBinding.value = response.data
    }
    return response.data
  }

  async function toggleEventBinding(projectId: number, bindingId: number, enabled: boolean) {
    const response = await api.patch(`/logic/projects/${projectId}/event-bindings/${bindingId}/toggle`, null, {
      params: { enabled }
    })
    const index = eventBindings.value.findIndex(b => b.id === bindingId)
    if (index !== -1) {
      eventBindings.value[index].enabled = enabled
    }
    return response.data
  }

  async function deleteEventBinding(projectId: number, bindingId: number) {
    await api.delete(`/logic/projects/${projectId}/event-bindings/${bindingId}`)
    eventBindings.value = eventBindings.value.filter(b => b.id !== bindingId)
    if (currentBinding.value?.id === bindingId) {
      currentBinding.value = null
    }
  }

  async function duplicateEventBinding(projectId: number, bindingId: number, newName?: string) {
    const params = newName ? { newName } : {}
    const response = await api.post(`/logic/projects/${projectId}/event-bindings/${bindingId}/duplicate`, null, { params })
    eventBindings.value.push(response.data)
    return response.data
  }

  async function validateEventBinding(projectId: number, actions: Action[], condition?: Condition) {
    const response = await api.post(`/logic/projects/${projectId}/event-bindings/validate`, { actions, condition })
    return response.data
  }

  async function fetchVariables(projectId: number, scope?: string) {
    const params = scope ? { scope } : {}
    const response = await api.get(`/logic/projects/${projectId}/variables`, { params })
    variables.value = response.data
  }

  async function createVariable(projectId: number, data: Omit<LogicVariable, 'id' | 'createdAt' | 'updatedAt'>) {
    const response = await api.post(`/logic/projects/${projectId}/variables`, data)
    variables.value.push(response.data)
    return response.data
  }

  async function updateVariable(projectId: number, variableId: number, data: Partial<LogicVariable>) {
    const response = await api.put(`/logic/projects/${projectId}/variables/${variableId}`, data)
    const index = variables.value.findIndex(v => v.id === variableId)
    if (index !== -1) {
      variables.value[index] = response.data
    }
    return response.data
  }

  async function deleteVariable(projectId: number, variableId: number) {
    await api.delete(`/logic/projects/${projectId}/variables/${variableId}`)
    variables.value = variables.value.filter(v => v.id !== variableId)
  }

  async function fetchExecutionLogs(projectId: number, eventBindingId?: number, limit?: number) {
    const params: any = {}
    if (eventBindingId) params.eventBindingId = eventBindingId
    if (limit) params.limit = limit
    const response = await api.get(`/logic/projects/${projectId}/execution-logs`, { params })
    executionLogs.value = response.data
  }

  function createEmptyBinding(overrides?: Partial<EventBinding>): EventBinding {
    return {
      eventName: 'click',
      eventSource: 'component' as EventSource,
      actions: [],
      enabled: true,
      sortOrder: 0,
      ...overrides
    }
  }

  function createEmptyAction(type: ActionType): Action {
    return {
      id: `action_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`,
      type,
      config: getDefaultConfig(type)
    }
  }

  function getDefaultConfig(type: ActionType) {
    switch (type) {
      case 'call_api':
        return { apiId: '', params: {}, method: 'GET' }
      case 'set_data':
        return { target: '', value: '', mode: 'set' }
      case 'navigate':
        return { params: {}, type: 'push' }
      case 'show_message':
        return { content: '', type: 'info' as const }
      case 'set_storage':
        return { key: '', value: '', storageType: 'local' as const }
      case 'get_storage':
        return { key: '', target: '', storageType: 'local' as const }
      case 'remove_storage':
        return { key: '', storageType: 'local' as const }
      case 'condition':
        return { condition: { type: 'simple' as const }, trueActions: [], falseActions: [] }
      case 'loop':
        return { source: '', itemName: 'item', actions: [] }
      case 'delay':
        return { duration: 1000 }
      case 'console_log':
        return { level: 'log' as const, message: '' }
      case 'set_component_prop':
        return { componentId: '', propName: '', value: '' }
      case 'trigger_event':
        return { eventName: '' }
      case 'confirm':
        return { content: '', confirmActions: [] }
      default:
        return {}
    }
  }

  function createEmptyCondition(): Condition {
    return {
      type: 'simple',
      left: '',
      operator: '==',
      right: ''
    }
  }

  return {
    eventBindings,
    variables,
    transformers,
    executionLogs,
    currentBinding,
    loading,
    enabledBindings,
    fetchEventBindings,
    getEventBinding,
    createEventBinding,
    updateEventBinding,
    toggleEventBinding,
    deleteEventBinding,
    duplicateEventBinding,
    validateEventBinding,
    fetchVariables,
    createVariable,
    updateVariable,
    deleteVariable,
    fetchExecutionLogs,
    createEmptyBinding,
    createEmptyAction,
    getDefaultConfig,
    createEmptyCondition
  }
})
```
