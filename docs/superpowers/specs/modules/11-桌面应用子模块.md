# 子模块 11: 桌面应用子模块设计文档

## 概述

基于Electron + Node.js的桌面应用，提供离线功能。桌面应用将复用Web端的Vue 3代码，通过IPC通信与主进程交互，实现本地数据存储、项目导入导出、自动更新等功能。

### 核心功能

- 离线项目管理（MySQL本地存储）
- 云同步功能（支持与Web端同步）
- 项目导入/导出（ZIP格式）
- 自动更新（electron-updater）
- 多窗口支持（主编辑器、预览窗口）
- 系统托盘和快捷操作
- 原生菜单和快捷键
- 文件系统访问
- 预览功能（多设备模拟）
- 代码生成（本地运行）
- 主题支持（亮色/暗色）

## 技术栈

- Electron (主进程)
- Vue 3 (渲染进程，复用Web端代码)
- TypeScript
- MySQL 8.0+ (本地数据库)
- Prisma (ORM)
- electron-updater (自动更新)
- electron-store (本地设置存储)
- adm-zip (ZIP导入导出)
- got (HTTP请求)
- Vite (构建工具)

### 数据库初始化和迁移

```typescript
// main/database/db.ts
import Database from "@prisma/client";
import * as path from "path";
import { app } from "electron";
import { migrate } from "./migrations";

let db: Database.Database | null = null;

export function initDatabase(): Database.Database {
  if (db) return db;

  const userDataPath = app.getPath("userData");
  const dbPath = path.join(userDataPath, "nocode.db");

  db = new Database(dbPath);
  db.pragma("journal_mode = WAL");

  // 执行迁移
  migrate(db);

  return db;
}

export function getDb(): Database.Database {
  if (!db) {
    return initDatabase();
  }
  return db;
}

export function closeDb(): void {
  if (db) {
    db.close();
    db = null;
  }
}

// main/database/migrations.ts
import Database from "@prisma/client";

const MIGRATIONS = [
  // 初始版本
  {
    version: 1,
    up: (db: Database.Database) => {
      db.exec(`
        CREATE TABLE IF NOT EXISTS local_projects (
          id TEXT PRIMARY KEY,
          name TEXT NOT NULL,
          description TEXT,
          thumbnail TEXT,
          data TEXT NOT NULL,
          is_synced INTEGER DEFAULT 0,
          cloud_id TEXT,
          last_synced_at TEXT,
          created_at TEXT DEFAULT CURRENT_TIMESTAMP,
          updated_at TEXT DEFAULT CURRENT_TIMESTAMP
        );

        CREATE INDEX IF NOT EXISTS idx_local_projects_updated
          ON local_projects(updated_at DESC);
        CREATE INDEX IF NOT EXISTS idx_local_projects_synced
          ON local_projects(is_synced);

        CREATE TABLE IF NOT EXISTS project_snapshots (
          id TEXT PRIMARY KEY,
          project_id TEXT NOT NULL,
          data TEXT NOT NULL,
          created_at TEXT DEFAULT CURRENT_TIMESTAMP,
          FOREIGN KEY (project_id) REFERENCES local_projects(id) ON DELETE CASCADE
        );

        CREATE INDEX IF NOT EXISTS idx_snapshots_project
          ON project_snapshots(project_id);

        CREATE TABLE IF NOT EXISTS sync_queue (
          id INTEGER PRIMARY KEY AUTOINCREMENT,
          project_id TEXT NOT NULL,
          action TEXT NOT NULL,
          data TEXT,
          created_at TEXT DEFAULT CURRENT_TIMESTAMP,
          retry_count INTEGER DEFAULT 0
        );

        CREATE TABLE IF NOT EXISTS settings (
          key TEXT PRIMARY KEY,
          value TEXT,
          updated_at TEXT DEFAULT CURRENT_TIMESTAMP
        );

        CREATE TABLE IF NOT EXISTS recent_files (
          id INTEGER PRIMARY KEY AUTOINCREMENT,
          project_id TEXT NOT NULL,
          project_name TEXT NOT NULL,
          file_path TEXT,
          opened_at TEXT DEFAULT CURRENT_TIMESTAMP
        );

        CREATE INDEX IF NOT EXISTS idx_recent_files_opened
          ON recent_files(opened_at DESC);
      `);
    },
  },
  // 版本2：添加更多字段
  {
    version: 2,
    up: (db: Database.Database) => {
      // 未来的迁移
    },
  },
];

export function migrate(db: Database.Database): void {
  // 获取当前版本
  const versionResult = db.prepare("PRAGMA user_version").get() as {
    user_version: number;
  };
  let currentVersion = versionResult.user_version;

  // 执行迁移
  for (const migration of MIGRATIONS) {
    if (migration.version > currentVersion) {
      const transaction = db.transaction(() => {
        migration.up(db);
        db.pragma(`user_version = ${migration.version}`);
      });
      transaction();
      currentVersion = migration.version;
    }
  }
}
```

### 项目数据访问层

```typescript
// main/project-manager.ts
import Database from "@prisma/client";
import { getDb } from "./database/sqlite";
import type { LocalProject, ProjectSnapshot } from "../shared/types";
import * as crypto from "crypto";

export class ProjectManager {
  private db: Database.Database;

  constructor() {
    this.db = getDb();
  }

  // 创建项目
  createProject(
    data: Omit<LocalProject, "created_at" | "updated_at">,
  ): LocalProject {
    const now = new Date().toISOString();
    const stmt = this.db.prepare(`
      INSERT INTO local_projects (
        id, name, description, thumbnail, data,
        is_synced, cloud_id, last_synced_at, created_at, updated_at
      ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
    `);

    const project: LocalProject = {
      ...data,
      created_at: now,
      updated_at: now,
    };

    stmt.run(
      project.id,
      project.name,
      project.description,
      project.thumbnail,
      project.data,
      project.is_synced ? 1 : 0,
      project.cloud_id,
      project.last_synced_at,
      project.created_at,
      project.updated_at,
    );

    // 创建快照
    this.createSnapshot(project.id, project.data);

    return project;
  }

  // 更新项目
  updateProject(
    id: string,
    data: Partial<
      Pick<
        LocalProject,
        | "name"
        | "description"
        | "thumbnail"
        | "data"
        | "is_synced"
        | "cloud_id"
        | "last_synced_at"
      >
    >,
  ): LocalProject | null {
    const now = new Date().toISOString();

    const updates: string[] = [];
    const params: any[] = [];

    Object.entries(data).forEach(([key, value]) => {
      if (key === "is_synced") {
        updates.push("is_synced = ?");
        params.push(value ? 1 : 0);
      } else {
        updates.push(`${key} = ?`);
        params.push(value);
      }
    });

    if (updates.length === 0) return this.getProject(id);

    updates.push("updated_at = ?");
    params.push(now);
    params.push(id);

    const stmt = this.db.prepare(`
      UPDATE local_projects
      SET ${updates.join(", ")}
      WHERE id = ?
    `);

    stmt.run(...params);

    // 为数据变更创建快照
    if (data.data) {
      this.createSnapshot(id, data.data);
    }

    return this.getProject(id);
  }

  // 获取项目
  getProject(id: string): LocalProject | null {
    const stmt = this.db.prepare("SELECT * FROM local_projects WHERE id = ?");
    const row = stmt.get(id) as any;
    if (!row) return null;

    return {
      ...row,
      is_synced: !!row.is_synced,
    };
  }

  // 列出项目
  listProjects(limit?: number, offset?: number): LocalProject[] {
    let sql = "SELECT * FROM local_projects ORDER BY updated_at DESC";
    const params: any[] = [];

    if (limit) {
      sql += " LIMIT ?";
      params.push(limit);
      if (offset) {
        sql += " OFFSET ?";
        params.push(offset);
      }
    }

    const stmt = this.db.prepare(sql);
    const rows = stmt.all(...params) as any[];

    return rows.map((row) => ({
      ...row,
      is_synced: !!row.is_synced,
    }));
  }

  // 删除项目
  deleteProject(id: string): boolean {
    const transaction = this.db.transaction(() => {
      // 删除快照
      this.db
        .prepare("DELETE FROM project_snapshots WHERE project_id = ?")
        .run(id);
      // 删除项目
      const result = this.db
        .prepare("DELETE FROM local_projects WHERE id = ?")
        .run(id);
      return result.changes > 0;
    });
    return transaction();
  }

  // 创建快照
  private createSnapshot(projectId: string, data: string): void {
    const id = crypto.randomUUID();
    this.db
      .prepare(
        `
      INSERT INTO project_snapshots (id, project_id, data)
      VALUES (?, ?, ?)
    `,
      )
      .run(id, projectId, data);

    // 只保留最近10个快照
    this.db
      .prepare(
        `
      DELETE FROM project_snapshots
      WHERE project_id = ? AND id NOT IN (
        SELECT id FROM project_snapshots
        WHERE project_id = ?
        ORDER BY created_at DESC
        LIMIT 10
      )
    `,
      )
      .run(projectId, projectId);
  }

  // 获取快照列表
  getSnapshots(projectId: string): ProjectSnapshot[] {
    const stmt = this.db.prepare(`
      SELECT * FROM project_snapshots
      WHERE project_id = ?
      ORDER BY created_at DESC
    `);
    return stmt.all(projectId) as ProjectSnapshot[];
  }

  // 恢复到快照
  restoreSnapshot(snapshotId: string): LocalProject | null {
    const snapshotStmt = this.db.prepare(`
      SELECT * FROM project_snapshots WHERE id = ?
    `);
    const snapshot = snapshotStmt.get(snapshotId) as ProjectSnapshot;

    if (!snapshot) return null;

    return this.updateProject(snapshot.project_id, {
      data: snapshot.data,
    });
  }

  // 最近文件操作
  addRecentFile(
    projectId: string,
    projectName: string,
    filePath?: string,
  ): void {
    this.db
      .prepare(
        `
      INSERT INTO recent_files (project_id, project_name, file_path)
      VALUES (?, ?, ?)
    `,
      )
      .run(projectId, projectName, filePath);

    // 只保留最近20个
    this.db
      .prepare(
        `
      DELETE FROM recent_files
      WHERE id NOT IN (
        SELECT id FROM recent_files ORDER BY opened_at DESC LIMIT 20
      )
    `,
      )
      .run();
  }

  getRecentFiles(limit: number = 10): any[] {
    return this.db
      .prepare(
        `
      SELECT * FROM recent_files ORDER BY opened_at DESC LIMIT ?
    `,
      )
      .all(limit) as any[];
  }
}

// 单例
let projectManager: ProjectManager | null = null;
export function getProjectManager(): ProjectManager {
  if (!projectManager) {
    projectManager = new ProjectManager();
  }
  return projectManager;
}
```

```typescript
// main/settings.ts
import Database from "@prisma/client";
import { getDb } from "./database/sqlite";
import type { Setting } from "../shared/types";

export class SettingsManager {
  private db: Database.Database;
  private cache: Map<string, any> = new Map();

  constructor() {
    this.db = getDb();
    this.loadAll();
  }

  private loadAll(): void {
    const stmt = this.db.prepare("SELECT * FROM settings");
    const rows = stmt.all() as Setting[];
    rows.forEach((row) => {
      try {
        this.cache.set(row.key, JSON.parse(row.value));
      } catch {
        this.cache.set(row.key, row.value);
      }
    });
  }

  get<T>(key: string, defaultValue?: T): T {
    if (this.cache.has(key)) {
      return this.cache.get(key) as T;
    }
    return defaultValue as T;
  }

  set(key: string, value: any): void {
    const valueStr = typeof value === "string" ? value : JSON.stringify(value);
    const now = new Date().toISOString();

    this.db
      .prepare(
        `
      INSERT OR REPLACE INTO settings (key, value, updated_at)
      VALUES (?, ?, ?)
    `,
      )
      .run(key, valueStr, now);

    this.cache.set(key, value);
  }

  getAll(): Record<string, any> {
    const result: Record<string, any> = {};
    this.cache.forEach((value, key) => {
      result[key] = value;
    });
    return result;
  }
}

let settingsManager: SettingsManager | null = null;
export function getSettingsManager(): SettingsManager {
  if (!settingsManager) {
    settingsManager = new SettingsManager();
  }
  return settingsManager;
}
```

```typescript
  app.quit();

  app.on("second-instance", () => {

    if (mainWindow) {
      if (mainWindow.isMinimized()) mainWindow.restore();
      mainWindow.focus();

      try {
        await installExtension(VUEJS3_DEVTOOLS);
        mainWindow.webContents.openDevTools();
      } catch (e) {
        console.error("Failed to install devtools:", e);
      }

      mainWindow.loadURL("http://localhost:5173");
    } else {
      mainWindow.loadFile(path.join(__dirname, "../renderer/index.html"));
    }
  }

  app.whenReady().then(async () => {
    // 初始化数据库
    initDatabase();

    // 设置
    getSettingsManager();

    // 菜单
    setupMenu();

    // IPC
    setupIpcHandlers();

    // 托盘
    setupTray();

      if (BrowserWindow.getAllWindows().length === 0) {
        createWindow();
      }
    });

    if (process.platform !== "darwin") {
      app.quit();

      e.preventDefault();
      isQuitting = true;
```

```typescript
// main/window-manager.ts
import { BrowserWindow, screen, app } from "electron";
import * as path from "path";
import { getSettingsManager } from "./settings";

let mainWindow: BrowserWindow | null = null;
let previewWindow: BrowserWindow | null = null;

export function createMainWindow(): BrowserWindow {
  const settings = getSettingsManager();
  const savedBounds = settings.get<Electron.Rectangle>("windowBounds");

  const { width, height } = screen.getPrimaryDisplay().workAreaSize;

  const win = new BrowserWindow({
    x: savedBounds?.x,
    y: savedBounds?.y,
    width: savedBounds?.width || Math.min(1400, width),
    height: savedBounds?.height || Math.min(900, height),
    minWidth: 1200,
    minHeight: 700,
    show: false,
    webPreferences: {
      preload: path.join(__dirname, "../preload/preload.js"),
      nodeIntegration: false,
      contextIsolation: true,
      sandbox: false,
      spellcheck: false,
    },
    titleBarStyle: "hiddenInset",
    frame: process.platform === "darwin" ? false : true,
    backgroundColor: "#1a1a2e",
    trafficLightPosition: { x: 18, y: 18 },
  });

  mainWindow = win;

  // 保存窗口位置和大小
  win.on("close", () => {
    const bounds = win.getBounds();
    settings.set("windowBounds", bounds);
  });

  win.on("closed", () => {
    mainWindow = null;
  });

  win.once("ready-to-show", () => {
    win.show();
  });

  return win;
}

export function getMainWindow(): BrowserWindow | null {
  return mainWindow;
}

export function createPreviewWindow(

    win.loadURL(options.url);
  }

  win.once("ready-to-show", () => {
    win.show();
  });

  win.on("closed", () => {
    previewWindow = null;
  });

  return win;
}

export function getPreviewWindow(): BrowserWindow | null {
  return previewWindow;
}

export function sendToMainWindow(channel: string, ...args: any[]): void {
  if (mainWindow && !mainWindow.isDestroyed()) {
    mainWindow.webContents.send(channel, ...args);
  }
}

  BrowserWindow.getAllWindows().forEach((win) => {
    if (!win.isDestroyed()) {
      win.webContents.send(channel, ...args);
    }
  });
}
```

```typescript
import { getProjectManager } from "./project-manager";
import { getSettingsManager } from "./settings";
import Database from "@prisma/client";
import got from "got";

type SyncStatus = "idle" | "syncing" | "error";

export class SyncService {
  private db: Database.Database;
  private status: SyncStatus = "idle";
  private lastError?: string;
  private syncInterval?: NodeJS.Timeout;
  private projectManager: ReturnType<typeof getProjectManager>;

  // 启动自动同步
  startAutoSync(intervalMs: number = 5 * 60 * 1000): void {
    if (this.syncInterval) {
      clearInterval(this.syncInterval);
    }

    this.syncInterval = setInterval(() => {
      this.syncAll().catch(console.error);
    }, intervalMs);
  }

  // 停止自动同步
  stopAutoSync(): void {
    if (this.syncInterval) {
      clearInterval(this.syncInterval);
      this.syncInterval = undefined;
    }
  }

  async syncAll(): Promise<void> {
    if (this.status === "syncing") return;

    const apiBase = this.settingsManager.get<string>("apiBase");
    const authToken = this.settingsManager.get<string>("authToken");

    if (!apiBase || !authToken) {
      return;

      console.error("Sync failed:", err);

      this.status = "idle";

      .prepare(
        `
      SELECT * FROM sync_queue ORDER BY created_at ASC
    `,
      )
      .all() as any[];

    for (const item of queue) {
      try {
        await this.processQueueItem(item);
        // 成功处理后删除
        this.db.prepare("DELETE FROM sync_queue WHERE id = ?").run(item.id);
      } catch (err) {

          .prepare(
            `
          UPDATE sync_queue SET retry_count = retry_count + 1
          WHERE id = ?
        `,
          )
          .run(item.id);

        // 如果重试次数过多，标记失败
        if (item.retry_count >= 5) {
          this.db.prepare("DELETE FROM sync_queue WHERE id = ?").run(item.id);
        }
      }
    }
```

```typescript
// renderer/src/composables/useAutoUpdater.ts
import { ref, computed } from "vue";
import { useDesktopAPI } from "./useDesktopAPI";

type UpdateStatus =
  | "idle"
  | "checking"
  | "available"
  | "downloading"
  | "downloaded"
  | "error";

export function useAutoUpdater() {
  const { isDesktop, on, updaterCheck, updaterDownload, updaterInstall } =
    useDesktopAPI();

  const status = ref<UpdateStatus>("idle");
  const progress = ref(0);
  const error = ref<string>();
  const updateInfo = ref<any>();

  if (isDesktop) {
    on("updater:checking", () => {
      status.value = "checking";
    });

    on("updater:available", (info) => {
      status.value = "available";
      updateInfo.value = info;
    });

    on("updater:not-available", () => {
      status.value = "idle";
    });

    on("updater:error", (err) => {
      status.value = "error";
      error.value = err.message;
    });

    on("updater:progress", (p) => {
      progress.value = p.percent;
    });

    on("updater:downloaded", (info) => {
      status.value = "downloaded";
      updateInfo.value = info;
    });
  }

  const checkForUpdates = () => {
    if (isDesktop) {
      updaterCheck();
    }
  };

  const download = () => {
    if (isDesktop) {
      status.value = "downloading";
      updaterDownload();
    }
  };

  const install = () => {
    if (isDesktop) {
      updaterInstall();
    }
  };

  const canCheck = computed(
    () => isDesktop && ["idle", "available"].includes(status.value),
  );
  const canDownload = computed(() => isDesktop && status.value === "available");
  const canInstall = computed(() => isDesktop && status.value === "downloaded");

  return {
    status,
    progress,
    error,
    updateInfo,
    checkForUpdates,
    download,
    install,
    canCheck,
    canDownload,
    canInstall,
  };
}
```
