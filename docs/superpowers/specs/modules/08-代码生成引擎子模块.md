## 生成器基类

```typescript
// src/base/generator.ts

import {
  Project,
  Page,
  ApiConfig,
  DataModel,
  GeneratedFile,
  GenerationResult,
  GenerationOptions,
  PlatformInfo
} from './types'

// 代码生成器抽象基类
export abstract class BaseCodeGenerator {
  // 平台ID，必须唯一
  abstract readonly platform: string
  // 平台名称
  abstract readonly name: string
  // 平台信息
  abstract readonly platformInfo: PlatformInfo

  protected project: Project
  protected options: GenerationOptions

  // 生成器构造函数
  constructor(project: Project, options?: GenerationOptions) {
    this.project = project
    this.options = {
      useTypeScript: true,
      includeComments: true,
      includeTests: false,
      includeLintConfig: false,
      ...options
    }
  }

  // 生成完整项目代码
  abstract generate(): Promise<GenerationResult>

  // 生成单个页面
  abstract generatePage(page: Page): Promise<GeneratedFile[]>

  // 生成API服务
  abstract generateApiService(apis: ApiConfig[]): Promise<GeneratedFile[]>

  // 生成数据模型
  abstract generateModels(models: DataModel[]): Promise<GeneratedFile[]>

  // 生成项目配置文件
  abstract generateProjectConfig(): Promise<GeneratedFile[]>

  // 生成README文档
  abstract generateReadme(): Promise<GeneratedFile>

  // 检查项目是否完整
  protected validateProject(): { valid: boolean; errors: string[] } {
    const errors: string[] = []

    if (!this.project.name) {
      errors.push('项目名称不能为空')
    }

    if (!this.project.pages || this.project.pages.length === 0) {
      errors.push('项目至少需要一个页面')
    }

    const homePages = this.project.pages.filter(p => p.isHome)
    if (homePages.length === 0) {
      errors.push('需要设置一个首页')
    }

    return {
      valid: errors.length === 0,
      errors
    }
  }

  // 创建生成结果
  protected createResult(
    files: GeneratedFile[],
    warnings: string[] = [],
    errors: string[] = [],
    startTime: number
  ): GenerationResult {
    const totalLines = files.reduce((sum, file) => {
      return sum + file.content.split('\n').length
    }, 0)

    const totalSize = files.reduce((sum, file) => {
      return sum + new Blob([file.content]).size
    }, 0)

    return {
      success: errors.length === 0,
      files,
      warnings,
      errors,
      statistics: {
        totalFiles: files.length,
        totalLines,
        totalSize,
        generationTime: Date.now() - startTime
      }
    }
  }

  // 生成文件头注释
  protected generateFileHeader(description: string): string {
    if (!this.options.includeComments) {
      return ''
    }

    const now = new Date().toISOString()
    const lang = this.options.useTypeScript ? 'TypeScript' : 'JavaScript'

    return `/**
 * ${description}
 * Generated by NoCode App Platform
 * Generated at: ${now}
 * Language: ${lang}
 */

`
  }
}
```

```typescript
// src/base/plugin-manager.ts

import { BaseCodeGenerator } from './generator'
import { Project, GenerationOptions, PlatformInfo } from './types'

export class CodeGeneratorPluginManager {
  private generators: Map<string, new (project: Project, options?: GenerationOptions) => BaseCodeGenerator> =
    new Map()
  private platformInfos: Map<string, PlatformInfo> = new Map()

  // 注册生成器
  registerGenerator(
    platform: string,
    ctor: new (project: Project, options?: GenerationOptions) => BaseCodeGenerator,
    platformInfo: PlatformInfo
  ): void {
    this.generators.set(platform, ctor)
    this.platformInfos.set(platform, platformInfo)
    console.log(`Registered generator for platform: ${platform}`)
  }

  // 获取生成器实例
  getGenerator(
    platform: string,
    project: Project,
    options?: GenerationOptions
  ): BaseCodeGenerator | undefined {
    const Ctor = this.generators.get(platform)
    if (Ctor) {
      return new Ctor(project, options)
    }
    return undefined
  }

  // 获取可用平台列表
  getAvailablePlatforms(): string[] {
    return Array.from(this.generators.keys())
  }

  // 获取平台信息
  getPlatformInfo(platform: string): PlatformInfo | undefined {
    return this.platformInfos.get(platform)
  }

  // 获取所有平台信息
  getAllPlatformInfos(): PlatformInfo[] {
    return Array.from(this.platformInfos.values())
  }

  // 检查平台是否支持
  isPlatformSupported(platform: string): boolean {
    return this.generators.has(platform)
  }
}

// 单例实例
export const pluginManager = new CodeGeneratorPluginManager()
```

```typescript
// src/base/template-engine.ts

import * as Handlebars from 'handlebars'
import * as fs from 'fs'
import * as path from 'path'

export class TemplateEngine {
  private templates: Map<string, Handlebars.TemplateDelegate> = new Map()
  private templateDir: string
  private partials: Map<string, string> = new Map()

  constructor(templateDir: string) {
    this.templateDir = templateDir
    this.registerHelpers()
  }

  // 注册助手函数
  private registerHelpers(): void {
    // 转驼峰
    Handlebars.registerHelper('camelCase', (str: string) => {
      return str
        .split(/[-_\s]+/)
        .map((word, i) =>
          i === 0
            ? word.toLowerCase()
            : word.charAt(0).toUpperCase() + word.slice(1)
        )
        .join('')
    })

    // 转帕斯卡
    Handlebars.registerHelper('pascalCase', (str: string) => {
      return str
        .split(/[-_\s]+/)
        .map(word => word.charAt(0).toUpperCase() + word.slice(1))
        .join('')
    })

    // 转下划线
    Handlebars.registerHelper('snakeCase', (str: string) => {
      return str
        .split(/[-_\s]+/)
        .map(word => word.toLowerCase())
        .join('_')
    })

    // JSON格式化
    Handlebars.registerHelper('json', (obj: any, indent: number = 2) => {
      return JSON.stringify(obj, null, indent)
    })

    // 条件判断
    Handlebars.registerHelper('ifEquals', (a: any, b: any, options: any) => {
      if (a === b) {
        return options.fn(this)
      }
      return options.inverse(this)
    })

    // 安全字符串
    Handlebars.registerHelper('safeString', (str: string) => {
      return new Handlebars.SafeString(str)
    })
  }

  // 加载模板目录
  async loadTemplates(): Promise<void> {
    if (!fs.existsSync(this.templateDir)) {
      console.warn(`Template directory not found: ${this.templateDir}`)
      return
    }

    const files = fs.readdirSync(this.templateDir)

    for (const file of files) {
      const fullPath = path.join(this.templateDir, file)
      const stat = fs.statSync(fullPath)

      if (stat.isDirectory()) {
        await this.loadTemplateDir(fullPath)
      } else if (file.endsWith('.template')) {
        const templateName = file.replace('.template', '')
        const content = fs.readFileSync(fullPath, 'utf-8')
        const template = Handlebars.compile(content)
        this.templates.set(templateName, template)
      }
    }
  }

  // 递归加载子目录
  private async loadTemplateDir(dir: string): Promise<void> {
    const files = fs.readdirSync(dir)

    for (const file of files) {
      const fullPath = path.join(dir, file)
      const stat = fs.statSync(fullPath)

      if (stat.isDirectory()) {
        await this.loadTemplateDir(fullPath)
      } else if (file.endsWith('.template')) {
        const relPath = path.relative(this.templateDir, fullPath)
        const templateName = relPath.replace('.template', '').replace(/\\/g, '/')
        const content = fs.readFileSync(fullPath, 'utf-8')
        const template = Handlebars.compile(content)
        this.templates.set(templateName, template)
      } else if (file.endsWith('.partial')) {
        const partialName = file.replace('.partial', '')
        const content = fs.readFileSync(fullPath, 'utf-8')
        Handlebars.registerPartial(partialName, content)
        this.partials.set(partialName, content)
      }
    }
  }

  // 渲染模板
  render(templateName: string, context: any): string {
    const template = this.templates.get(templateName)
    if (!template) {
      throw new Error(`Template not found: ${templateName}`)
    }
    return template(context)
  }

  // 检查模板是否存在
  hasTemplate(templateName: string): boolean {
    return this.templates.has(templateName)
  }

  // 注册自定义模板
  registerTemplate(name: string, content: string): void {
    const template = Handlebars.compile(content)
    this.templates.set(name, template)
  }
}
```

```typescript
// src/mappers/component-mapper.ts

import { ComponentInstance } from '../base/types'

export interface MappedComponent {
  tagName: string
  attributes: Record<string, string>
  content: string
  children: MappedComponent[]
}

export abstract class ComponentMapper {
  protected project: any

  constructor(project: any) {
    this.project = project
  }

  // 映射组件
  abstract mapComponent(component: ComponentInstance): MappedComponent

  // 批量映射组件
  mapComponents(components: ComponentInstance[]): MappedComponent[] {
    return components.map(c => this.mapComponent(c))
  }

  // 生成组件代码
  abstract generateComponentCode(component: MappedComponent): string

  // 批量生成组件代码
  generateComponentsCode(components: MappedComponent[]): string {
    return components.map(c => this.generateComponentCode(c)).join('\n')
  }

  // 构建属性字符串
  protected buildAttributes(attributes: Record<string, string>): string {
    const attrs = Object.entries(attributes)
      .map(([key, value]) => `${key}="${value}"`)
      .join(' ')
    return attrs ? ' ' + attrs : ''
  }
}
```

```typescript
// src/plugins/wechat-miniapp/index.ts

import { BaseCodeGenerator } from '../../base/generator'
import {
  Project,
  Page,
  ApiConfig,
  DataModel,
  GeneratedFile,
  GenerationResult,
  PlatformInfo
} from '../../base/types'
import { TemplateEngine } from '../../base/template-engine'
import { WechatComponentMapper } from './wechat-component-mapper'
import { NamingUtils } from '../../utils/naming'

export class WechatMiniappGenerator extends BaseCodeGenerator {
  readonly platform = 'wechat_miniapp'
  readonly name = '微信小程序'
  readonly platformInfo: PlatformInfo = {
    id: 'wechat_miniapp',
    name: '微信小程序',
    language: 'TypeScript',
    framework: 'WXML + WXSS',
    fileExtension: '.ts'
  }

  private templateEngine: TemplateEngine
  private componentMapper: WechatComponentMapper

  constructor(project: Project, options?: any) {
    super(project, options)
    this.templateEngine = new TemplateEngine(__dirname + '/templates')
    this.componentMapper = new WechatComponentMapper(project)
  }

  // 初始化
  async initialize(): Promise<void> {
    await this.templateEngine.loadTemplates()
  }

  async generate(): Promise<GenerationResult> {
    const startTime = Date.now()
    const files: GeneratedFile[] = []
    const warnings: string[] = []
    const errors: string[] = []

    try {
      // 验证项目
      const validation = this.validateProject()
      if (!validation.valid) {
        errors.push(...validation.errors)
        return this.createResult(files, warnings, errors, startTime)
      }

      // 初始化模板引擎
      await this.initialize()

      // 生成 app.json
      files.push(this.generateAppJson())

      // 生成 app.ts
      files.push(this.generateAppTs())

      // 生成 app.wxss
      files.push(this.generateAppWxss())

      // 生成 sitemap.json
      files.push(this.generateSitemapJson())

      // 生成 project.config.json
      files.push(this.generateProjectConfigJson())

      // 生成各页面
      for (const page of this.project.pages || []) {
        const pageFiles = await this.generatePage(page)
        files.push(...pageFiles)
      }

      // 生成API服务
      if (this.project.apis && this.project.apis.length > 0) {
        const apiFiles = await this.generateApiService(this.project.apis)
        files.push(...apiFiles)
      }

      // 生成数据模型
      if (this.project.dataModels && this.project.dataModels.length > 0) {
        const modelFiles = await this.generateModels(this.project.dataModels)
        files.push(...modelFiles)
      }

      // 生成 package.json
      files.push(this.generatePackageJson())

      // 生成 tsconfig.json
      files.push(this.generateTsConfig())

      // 生成 README.md
      files.push(await this.generateReadme())

    } catch (error: any) {
      errors.push(error.message || '生成失败')
    }

    return this.createResult(files, warnings, errors, startTime)
  }

  // 生成 app.json
  private generateAppJson(): GeneratedFile {
    const pages = this.project.pages?.map(p => p.path) || []
    const window = {
      navigationBarTitleText: this.project.name,
      navigationBarBackgroundColor: '#ffffff',
      navigationBarTextStyle: 'black' as const
    }
    const tabBar = this.project.config?.tabBar || null

    const content = JSON.stringify({
      pages,
      window,
      tabBar,
      style: 'v2',
      sitemapLocation: 'sitemap.json'
    }, null, 2)

    return {
      path: 'app.json',
      content,
      language: 'json'
    }
  }

  // 生成 app.ts
  private generateAppTs(): GeneratedFile {
    const content = this.generateFileHeader('小程序入口文件') + `
App({
  onLaunch() {
    console.log('App Launch')
  },

  onShow() {
    console.log('App Show')
  },

  onHide() {
    console.log('App Hide')
  },

  globalData: {
    userInfo: null,
    projectName: '${this.project.name}'
  }
})
`
    return {
      path: 'app.ts',
      content,
      language: 'typescript'
    }
  }

  // 生成 app.wxss
  private generateAppWxss(): GeneratedFile {
    const content = `/* app.wxss */
/* Generated by NoCode App Platform */

page {
  background-color: #f5f5f5;
  font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', 'Roboto', 'Oxygen', 'Ubuntu', 'Cantarell', 'Fira Sans', 'Droid Sans', 'Helvetica Neue', sans-serif;

}

}
`
    return {
      path: 'app.wxss',
      content,
      language: 'css'
    }
  }

  private generateSitemapJson(): GeneratedFile {
    const content = JSON.stringify({
      desc: '关于本文件的更多信息，请参考文档 https://developers.weixin.qq.com/miniprogram/dev/framework/sitemap.html',
      rules: [{
        action: 'allow',
        page: '*'
      }]
    }, null, 2)

    return {
      path: 'sitemap.json',
      content,
      language: 'json'
    }
  }

  // 生成 project.config.json
  private generateProjectConfigJson(): GeneratedFile {
    const content = JSON.stringify({
      description: '项目配置文件',
      packOptions: {
        ignore: []
      },
      setting: {
        urlCheck: true,
        es6: true,
        enhance: true,
        postcss: true,
        preloadBackgroundData: false,
        minified: true,
        newFeature: true,
        coverView: true,
        nodeModules: false,

      },

    }, null, 2)

    return {

    }
  }

  async generatePage(page: Page): Promise<GeneratedFile[]> {
    const files: GeneratedFile[] = []

    // 生成 wxml
    const wxmlContent = this.generateWxml(page)
    files.push({
      path: `${page.path}/index.wxml`,
      content: wxmlContent,
      language: 'xml'

    // 生成 wxss
    const wxssContent = this.generateWxss(page)
    files.push({
      path: `${page.path}/index.wxss`,
      content: wxssContent,
      language: 'css'
    })

    // 生成 ts
```

```typescript
// src/plugins/wechat-miniapp/wechat-component-mapper.ts

import { ComponentMapper, MappedComponent } from '../../mappers/component-mapper'
import { ComponentInstance } from '../../base/types'
import { NamingUtils } from '../../utils/naming'

export class WechatComponentMapper extends ComponentMapper {
  mapComponent(component: ComponentInstance): MappedComponent {
    const attributes: Record<string, string> = {}
    let content = ''

    // 构建属性
    this.buildComponentAttributes(component, attributes)

    // 构建内容
    content = this.buildComponentContent(component)

    // 映射子组件
    const children = component.children?.map(c => this.mapComponent(c)) || []

    return {
      tagName: this.getTagName(component.type),
      attributes,
      content,
      children
    }
  }

  generateComponentCode(component: MappedComponent): string {
    const attrs = this.buildAttributes(component.attributes)
    const content = component.content || ''
    const childrenCode = component.children.length > 0
      ? '\n' + component.children.map(c => this.generateComponentCode(c)).join('\n') + '\n'
      : ''

    if (childrenCode) {
      return `<${component.tagName}${attrs}>${content}${childrenCode}</${component.tagName}>`
    } else if (content) {
      return `<${component.tagName}${attrs}>${content}</${component.tagName}>`
    } else {
      return `<${component.tagName}${attrs} />`
    }
  }

  private getTagName(type: string): string {
    const tagMap: Record<string, string> = {
      'button': 'button',
      'text': 'text',
      'image': 'image',
      'input': 'input',
      'textarea': 'textarea',
      'list': 'scroll-view',
      'card': 'view',
      'container': 'view',
      'form': 'form',
      'checkbox': 'checkbox',
      'radio': 'radio',
      'switch': 'switch',
      'slider': 'slider',
      'picker': 'picker',
      'swiper': 'swiper',
      'icon': 'icon',
      'progress': 'progress',
      'video': 'video',
      'audio': 'audio',
      'camera': 'camera',
      'map': 'map',
      'canvas': 'canvas'
    }
    return tagMap[type] || 'view'
  }

  private buildComponentAttributes(
    component: ComponentInstance,
    attributes: Record<string, string>
  ): void {
    // ID
    attributes['id'] = component.id

    // 组件特定属性
    switch (component.type) {
      case 'button':
        attributes['type'] = component.props.type || 'default'
        attributes['size'] = component.props.size || 'default'
        if (component.props.disabled) attributes['disabled'] = 'true'
        attributes['loading'] = String(component.props.loading || false)
        break

      case 'input':
        attributes['type'] = component.props.inputType || 'text'
        attributes['placeholder'] = component.props.placeholder || ''
        if (component.props.value) attributes['value'] = `{{ ${component.id}.value }}`
        attributes['maxlength'] = String(component.props.maxLength || 140)
        break

      case 'image':
        attributes['src'] = component.props.src || ''
        attributes['mode'] = component.props.mode || 'aspectFit'
        attributes['lazy-load'] = String(component.props.lazyLoad || false)
        break

      case 'list':
        attributes['scroll-y'] = 'true'
        if (component.props.scrollIntoView) {
          attributes['scroll-into-view'] = component.props.scrollIntoView
        }
        break

      case 'swiper':
        attributes['indicator-dots'] = String(component.props.indicatorDots || true)
        attributes['autoplay'] = String(component.props.autoplay || false)
        attributes['interval'] = String(component.props.interval || 5000)
        attributes['duration'] = String(component.props.duration || 500)
        break
    }

    // 绑定事件
    const bindings = this.getComponentBindings(component.id)
    for (const binding of bindings) {
      const eventName = this.toWechatEventName(binding.eventName)
      const handlerName = NamingUtils.camelCase(`${component.id}_${binding.eventName}`)
      attributes[`bind${eventName}`] = handlerName
    }
  }

  private buildComponentContent(component: ComponentInstance): string {
    switch (component.type) {
      case 'button':
        return component.props.text || '按钮'
      case 'text':
        return component.props.text || ''
      default:
        return ''
    }
  }

  private toWechatEventName(name: string): string {
    const map: Record<string, string> = {
      'click': 'tap',
      'longpress': 'longpress',
      'longtap': 'longtap',
      'submit': 'submit',
      'change': 'change',
      'focus': 'focus',
      'blur': 'blur',
      'input': 'input',
      'confirm': 'confirm',
      'getuserinfo': 'getuserinfo'
    }
    return map[name] || name
  }

  private getComponentBindings(componentId: string): any[] {
    const allBindings = this.project.eventBindings || []
    return allBindings.filter(b => b.componentId === componentId)
  }
}
```

```typescript
// src/utils/naming.ts

export class NamingUtils {
  // 驼峰命名
  static camelCase(str: string): string {
    return str
      .split(/[-_\s]+/)
      .map((word, i) =>
        i === 0
          ? word.toLowerCase()
          : word.charAt(0).toUpperCase() + word.slice(1)
      )
      .join('')
  }

  // 帕斯卡命名
  static pascalCase(str: string): string {
    return str
      .split(/[-_\s]+/)
      .map(word => word.charAt(0).toUpperCase() + word.slice(1))
      .join('')
  }

  // 下划线命名
  static snakeCase(str: string): string {
    return str
      .split(/[-_\s]+/)
      .map(word => word.toLowerCase())
      .join('_')
  }

  // 短横线命名
  static kebabCase(str: string): string {
    return str
      .split(/[-_\s]+/)
      .map(word => word.toLowerCase())
      .join('-')
  }

  // 大写常量命名
  static constantCase(str: string): string {
    return str
      .split(/[-_\s]+/)
      .map(word => word.toUpperCase())
      .join('_')
  }

  // 路径安全
  static pathSafe(str: string): string {
    return str
      .replace(/[^a-zA-Z0-9-_]/g, '-')
      .replace(/-+/g, '-')
      .toLowerCase()
  }
}
```

```typescript
  constructor() {
    this.registerGenerators()
  }

  private registerGenerators(): void {
    // 微信小程序
    pluginManager.registerGenerator(
      'wechat_miniapp',
      WechatMiniappGenerator,
      WechatMiniappGenerator.prototype.platformInfo
    )

    // H5
    pluginManager.registerGenerator(
      'h5',
      H5Generator,
      H5Generator.prototype.platformInfo
    )

    // 其他平台...
  }

  // 获取可用平台
  getAvailablePlatforms(): string[] {
    return pluginManager.getAvailablePlatforms()
  }

  // 获取平台信息
  getPlatformInfos(): any[] {
    return pluginManager.getAllPlatformInfos()
  }

  // 生成代码
  async generateCode(
    request: GenerationRequest,
    project: Project
  ): Promise<GenerationResult> {
    const platform = request.platform
    const options = request.options

    if (!pluginManager.isPlatformSupported(platform)) {
      return {
        success: false,
        files: [],
        warnings: [],
        errors: [`Platform not supported: ${platform}`]
      }
    }

    const generator = pluginManager.getGenerator(platform, project, options)
    if (!generator) {
      return {
        success: false,
        files: [],
        warnings: [],
        errors: ['Generator initialization failed']
      }
    }

    return await generator.generate()
  }
}
```

```typescript
// src/api/routes.ts

import { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify'
import { codeGeneratorService } from '../services/code-generator'
import { zipPackageService } from '../services/zip-package'

export async function setupApiRoutes(app: FastifyInstance): Promise<void> {
  // 获取可用平台
  app.get(
    '/api/code-generator/platforms',
    async (request: FastifyRequest, reply: FastifyReply) => {
      const platforms = codeGeneratorService.getPlatformInfos()
      reply.send({
        success: true,
        data: platforms
      })
    }
  )

  // 生成代码
  app.post(
    '/api/projects/:projectId/generate-code',
    async (request: FastifyRequest, reply: FastifyReply) => {
      const { projectId } = request.params as { projectId: string }
      const body = request.body as any

      // TODO: 从数据库获取项目
      const project = await fetchProject(projectId)

      const result = await codeGeneratorService.generateCode(
        {
          projectId,
          platform: body.platform,
          options: body.options
        },
        project
      )

      if (!result.success) {
        reply.status(400).send({
          success: false,
          error: result.errors.join(', ')
        })
        return
      }

      // 保存生成记录
      const generationId = await saveGenerationRecord(projectId, body.platform, result)

      reply.send({
        success: true,
        data: {
          generationId,
          files: result.files.map(f => ({
            path: f.path,
            size: f.size
          })),
          statistics: result.statistics,
          downloadUrl: `/api/generations/${generationId}/download`
        }
      })
    }
  )

  // 下载生成的代码
  app.get(
    '/api/generations/:generationId/download',
    async (request: FastifyRequest, reply: FastifyReply) => {
      const { generationId } = request.params as { generationId: string }

      const { project, generation } = await fetchGenerationData(generationId)

      const result = await codeGeneratorService.generateCode(
        {
          projectId: generation.projectId,
          platform: generation.targetPlatform
        },
        project
      )

        reply.status(400).send({
          success: false,
          error: 'Generation failed'
        })
        return
      }

      const zipBuffer = await zipPackageService.packageFiles(result.files)

      reply
        .type('application/zip')
        .header('Content-Disposition', `attachment; filename="${project.name}-${generation.targetPlatform}.zip"`)
        .send(zipBuffer)
    }
  )
}
```

```typescript
// src/services/zip-package.ts

import * as JSZip from 'jszip'
import { GeneratedFile } from '../base/types'

export class ZipPackageService {
  async packageFiles(files: GeneratedFile[]): Promise<Buffer> {
    const zip = new JSZip()

    for (const file of files) {
      zip.file(file.path, file.content)
    }

    return await zip.generateAsync({
      type: 'nodebuffer',
      compression: 'DEFLATE',
      compressionOptions: { level: 6 }
    })
  }

  async addFileToZip(
    zip: JSZip,
    path: string,
    content: string,
    replace: boolean = false
  ): Promise<void> {
    if (replace) {
      zip.remove(path)
    }
    zip.file(path, content)
  }
}
```
