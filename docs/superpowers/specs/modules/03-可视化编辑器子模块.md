# 子模块 03: 可视化编辑器子模块设计文档

## 概述

负责组件拖拽、布局编排、属性配置的前端可视化编辑器。

## 技术栈

- Vue 3 + TypeScript
- Pinia (状态管理)
- Tailwind CSS

## 组件库数据模型

### component_templates 表

| 字段 | 类型 | 约束 | 说明 |

## 核心功能实现

### 组件注册表

```typescript
// componentRegistry.ts
import type { ComponentDefinition } from '@/types/editor'

const componentRegistry = new Map<string, ComponentDefinition>()
const categoryOrder = ['基础', '表单', '列表', '导航', '展示', '布局']

export function registerComponent(def: ComponentDefinition) {
  if (componentRegistry.has(def.type)) {
    console.warn(`Component ${def.type} already registered, overwriting`)
  }
  componentRegistry.set(def.type, def)
}

export function registerComponents(defs: ComponentDefinition[]) {
  defs.forEach(registerComponent)
}

export function getComponent(type: string): ComponentDefinition | undefined {
  return componentRegistry.get(type)
}

export function getAllComponents(): ComponentDefinition[] {
  return Array.from(componentRegistry.values())
}

export function getComponentsByCategory(): Record<string, ComponentDefinition[]> {
  const result: Record<string, ComponentDefinition[]> = {}
  for (const comp of componentRegistry.values()) {
    if (!result[comp.category]) {
      result[comp.category] = []
    }
    result[comp.category].push(comp)
  }
  // 按指定顺序排序分类
  const sortedResult: Record<string, ComponentDefinition[]> = {}
  categoryOrder.forEach(cat => {
    if (result[cat]) {
      sortedResult[cat] = result[cat]
    }
  })
  // 添加其他分类
  Object.keys(result).forEach(cat => {
    if (!categoryOrder.includes(cat)) {
      sortedResult[cat] = result[cat]
    }
  })
  return sortedResult
}

export function getComponentCategories(): string[] {
  const categories = new Set<string>()
  for (const comp of componentRegistry.values()) {
    categories.add(comp.category)
  }
  return [...categories].sort((a, b) => {
    const aIndex = categoryOrder.indexOf(a)
    const bIndex = categoryOrder.indexOf(b)
    if (aIndex !== -1 && bIndex !== -1) return aIndex - bIndex
    if (aIndex !== -1) return -1
    if (bIndex !== -1) return 1
    return a.localeCompare(b)
  })
}

export function unregisterComponent(type: string): boolean {
  return componentRegistry.delete(type)
}

export function clearRegistry(): void {
  componentRegistry.clear()
}
```

### 拖拽逻辑

```typescript
// useDragDrop.ts
import { ref, computed } from 'vue'
import { useEditorStore } from '@/stores/editor'
import { getSnapPositions, snapToGrid } from '@/utils/snapUtils'

const DRAG_THRESHOLD = 5 // 像素，防止误触
const MIN_COMPONENT_SIZE = 10 // 最小组件尺寸

export function useDragDrop() {
  const store = useEditorStore()
  const isDragging = computed(() => store.dragState.isDragging)

  function onDragStartFromPanel(
    e: DragEvent,
    componentType: string
  ) {
    if (!e.dataTransfer) return

    e.dataTransfer.setData('componentType', componentType)
    e.dataTransfer.effectAllowed = 'copy'
    e.dataTransfer.dropEffect = 'copy'

    // 设置拖拽图像
    const ghost = createGhostElement(componentType)
    document.body.appendChild(ghost)
    e.dataTransfer.setDragImage(ghost, 0, 0)
    requestAnimationFrame(() => ghost.remove())
  }

  function onDragStartFromCanvas(
    e: DragEvent,
    componentId: string
  ) {
    if (!e.dataTransfer) return

    e.dataTransfer.setData('componentId', componentId)
    e.dataTransfer.effectAllowed = 'move'
    e.dataTransfer.dropEffect = 'move'

    const comp = store.components.find(c => c.id === componentId)
    if (!comp) return

    const rect = (e.currentTarget as HTMLElement).getBoundingClientRect()

    store.dragState = {
      isDragging: true,
      dragType: 'move',
      componentId,
      startX: e.clientX,
      startY: e.clientY,
      currentX: e.clientX,
      currentY: e.clientY,
      startPosition: JSON.parse(JSON.stringify(comp.position)),
    }
  }

  function onDragOverCanvas(e: DragEvent) {
    e.preventDefault()
    e.stopPropagation()

    if (!store.dragState.isDragging) return

    const rect = (e.currentTarget as HTMLElement).getBoundingClientRect()
    const x = e.clientX - rect.left
    const y = e.clientY - rect.top

    store.dragState.currentX = e.clientX
    store.dragState.currentY = e.clientY

    // 计算对齐辅助线
    if (store.canvasState.showGuides && store.dragState.componentId) {
      calculateGuideLines(x, y)
    }
  }

  function onDropOnCanvas(e: DragEvent) {
    e.preventDefault()
    e.stopPropagation()

    const componentType = e.dataTransfer?.getData('componentType')
    const componentId = e.dataTransfer?.getData('componentId')

    const rect = (e.currentTarget as HTMLElement).getBoundingClientRect()
    let x = (e.clientX - rect.left) / store.canvasState.zoom
    let y = (e.clientY - rect.top) / store.canvasState.zoom

    // 应用网格吸附
    if (store.canvasState.snapToGrid) {
      const snapped = snapToGrid(x, y, store.canvasState.gridSize)
      x = snapped.x
      y = snapped.y
    }

    if (componentType) {
      const def = getComponent(componentType)
      if (def) {
        const newId = store.addComponent(componentType, {
          type: 'absolute',
          x: x - (parseInt(def.defaultStyles.width || '100') / 2),
          y: y - (parseInt(def.defaultStyles.height || '40') / 2),
          width: def.defaultStyles.width,
          height: def.defaultStyles.height,
        })
        store.selectComponent(newId)
      }
    } else if (componentId && store.dragState.startPosition) {
      const dx = x - (store.dragState.startX - rect.left) / store.canvasState.zoom
      const dy = y - (store.dragState.startY - rect.top) / store.canvasState.zoom

      const newPos = JSON.parse(JSON.stringify(store.dragState.startPosition))
      if (newPos.type === 'absolute') {
        newPos.x = (newPos.x || 0) + dx
        newPos.y = (newPos.y || 0) + dy

        // 应用对齐吸附
        if (store.canvasState.showGuides) {
          const snapped = applySnapAlignment(componentId, newPos)
          newPos.x = snapped.x
          newPos.y = snapped.y
        }
      }

      store.moveComponent(componentId, newPos)
    }

    // 清除拖拽状态和辅助线
    store.dragState = {
      isDragging: false,
      dragType: 'new',
      startX: 0,
      startY: 0,
      currentX: 0,
      currentY: 0,
    }
    store.alignmentState.horizontalGuides = []
    store.alignmentState.verticalGuides = []
  }

  function calculateGuideLines(x: number, y: number) {
    const { snapThreshold } = store.alignmentState
    const componentId = store.dragState.componentId
    if (!componentId) return

    const draggedComp = store.components.find(c => c.id === componentId)
    if (!draggedComp || draggedComp.position.type !== 'absolute') return

    const draggedRect = {
      left: draggedComp.position.x || 0,
      top: draggedComp.position.y || 0,
      width: parseInt(draggedComp.styles.width || '100'),
      height: parseInt(draggedComp.styles.height || '100'),
    }
    draggedRect.right = draggedRect.left + draggedRect.width
    draggedRect.bottom = draggedRect.top + draggedRect.height
    draggedRect.centerX = draggedRect.left + draggedRect.width / 2
    draggedRect.centerY = draggedRect.top + draggedRect.height / 2

    const hGuides: GuideLine[] = []
    const vGuides: GuideLine[] = []

    // 检查与其他组件的对齐
    store.components.forEach(comp => {
      if (comp.id === componentId || comp.hidden) return
      if (comp.position.type !== 'absolute') return

      const rect = {
        left: comp.position.x || 0,
        top: comp.position.y || 0,
        width: parseInt(comp.styles.width || '100'),
        height: parseInt(comp.styles.height || '100'),
      }
      rect.right = rect.left + rect.width
      rect.bottom = rect.top + rect.height
      rect.centerX = rect.left + rect.width / 2
      rect.centerY = rect.top + rect.height / 2

      // 水平对齐
      ;[
        { pos: rect.left, type: 'left' as const },
        { pos: rect.centerX, type: 'center' as const },
        { pos: rect.right, type: 'right' as const },
      ].forEach(({ pos, type }) => {
        if (Math.abs(draggedRect.left - pos) < snapThreshold) {
          vGuides.push({ position: pos, type, componentId: comp.id })
        }
        if (Math.abs(draggedRect.centerX - pos) < snapThreshold) {
          vGuides.push({ position: pos, type, componentId: comp.id })
        }
        if (Math.abs(draggedRect.right - pos) < snapThreshold) {
          vGuides.push({ position: pos, type, componentId: comp.id })
        }
      })

      // 垂直对齐
      ;[
        { pos: rect.top, type: 'top' as const },
        { pos: rect.centerY, type: 'middle' as const },
        { pos: rect.bottom, type: 'bottom' as const },
      ].forEach(({ pos, type }) => {
        if (Math.abs(draggedRect.top - pos) < snapThreshold) {
          hGuides.push({ position: pos, type, componentId: comp.id })
        }
        if (Math.abs(draggedRect.centerY - pos) < snapThreshold) {
          hGuides.push({ position: pos, type, componentId: comp.id })
        }
        if (Math.abs(draggedRect.bottom - pos) < snapThreshold) {
          hGuides.push({ position: pos, type, componentId: comp.id })
        }
      })
    })

    store.alignmentState.horizontalGuides = hGuides
    store.alignmentState.verticalGuides = vGuides
  }

  function applySnapAlignment(componentId: string, pos: ComponentPosition) {
    const { snapThreshold } = store.alignmentState
    let { x = 0, y = 0 } = pos

    const comp = store.components.find(c => c.id === componentId)
    if (!comp || comp.position.type !== 'absolute') return { x, y }

    const width = parseInt(comp.styles.width || '100')
    const height = parseInt(comp.styles.height || '100')

    // 检查垂直辅助线吸附
    const draggedLeft = x
    const draggedRight = x + width
    const draggedCenterX = x + width / 2

    store.alignmentState.verticalGuides.forEach(guide => {
      if (Math.abs(draggedLeft - guide.position) < snapThreshold) {
        x = guide.position
      } else if (Math.abs(draggedRight - guide.position) < snapThreshold) {
        x = guide.position - width
      } else if (Math.abs(draggedCenterX - guide.position) < snapThreshold) {
        x = guide.position - width / 2
      }
    })

    // 检查水平辅助线吸附
    const draggedTop = y
    const draggedBottom = y + height
    const draggedCenterY = y + height / 2

    store.alignmentState.horizontalGuides.forEach(guide => {
      if (Math.abs(draggedTop - guide.position) < snapThreshold) {
        y = guide.position
      } else if (Math.abs(draggedBottom - guide.position) < snapThreshold) {
        y = guide.position - height
      } else if (Math.abs(draggedCenterY - guide.position) < snapThreshold) {
        y = guide.position - height / 2
      }
    })

    return { x, y }
  }

  function createGhostElement(componentType: string): HTMLElement {
    const def = getComponent(componentType)
    const ghost = document.createElement('div')
    ghost.style.cssText = `
      width: ${def?.defaultStyles.width || '100px'};
      height: ${def?.defaultStyles.height || '40px'};
      background: rgba(59, 130, 246, 0.3);
      border: 2px dashed #3b82f6;
      border-radius: 4px;
      pointer-events: none;
      opacity: 0.8;
    `
    return ghost
  }

  return {
    isDragging,
    onDragStartFromPanel,
    onDragStartFromCanvas,
    onDragOverCanvas,
    onDropOnCanvas,
  }
}
```

### 历史记录管理

```typescript
// useHistory.ts
import { computed } from 'vue'
import { useEditorStore } from '@/stores/editor'

const MAX_HISTORY = 100 // 最大历史记录数

export function useHistory() {
  const store = useEditorStore()

  const canUndo = computed(() => store.historyIndex > 0)
  const canRedo = computed(() => store.historyIndex < store.history.length)

  function undo() {
    if (!canUndo.value) return

    const entry = store.history[store.historyIndex - 1]
    if (!entry.undoData) return

    applyHistory(entry.undoData)
    store.historyIndex--
    store.isDirty = true
  }

  function redo() {
    if (!canRedo.value) return

    const entry = store.history[store.historyIndex]
    applyHistory(entry.data)
    store.historyIndex++
    store.isDirty = true
  }

  function applyHistory(data: any) {
    switch (data.type) {
      case 'add':
        if (data.components) {
          // 批量添加的反向是删除
          data.components.forEach((id: string) => {
            const idx = store.components.findIndex(c => c.id === id)
            if (idx !== -1) store.components.splice(idx, 1)
          })
        } else {
          const idx = store.components.findIndex(c => c.id === data.id)
          if (idx !== -1) store.components.splice(idx, 1)
        }
        break

      case 'delete':
        if (data.components) {
          data.components.forEach((comp: any) => {
            insertComponentAtPosition(comp)
          })
        } else {
          insertComponentAtPosition(data.component)
        }
        break

      case 'update':
        const comp = store.components.find(c => c.id === data.id)
        if (comp) {
          Object.assign(comp, data.updates)
        }
        break

      case 'move':
        const movedComp = store.components.find(c => c.id === data.id)
        if (movedComp) {
          movedComp.position = data.position
        }
        break

      case 'batch':
        data.operations.forEach((op: any) => {
          applyHistory(op)
        })
        break
    }
  }

  function insertComponentAtPosition(component: any) {
    if (component.parentId) {
      const parent = store.components.find(c => c.id === component.parentId)
      if (parent) {
        if (!parent.children) parent.children = []
        parent.children.splice(component.sortOrder, 0, component)
      }
    } else {
      store.components.splice(component.sortOrder, 0, component)
    }
  }

  function pushHistory(entry: {
    type: 'add' | 'update' | 'delete' | 'move' | 'batch'
    data: any
    undoData: any
    description?: string
  }) {
    // 清除当前位置之后的历史
    if (store.historyIndex < store.history.length) {
      store.history.splice(store.historyIndex)
    }

    store.history.push({
      id: generateId(),
      type: entry.type,
      timestamp: Date.now(),
      data: entry.data,
      undoData: entry.undoData,
      description: entry.description,
    })

    // 限制历史记录数量
    if (store.history.length > MAX_HISTORY) {
      store.history.shift()
    } else {
      store.historyIndex++
    }

    store.isDirty = true
  }

  function recordAdd(component: any, description?: string) {
    pushHistory({
      type: 'add',
      data: { id: component.id, component },
      undoData: { type: 'delete', id: component.id },
      description: description || `添加 ${component.type}`,
    })
  }

  function recordDelete(component: any, description?: string) {
    pushHistory({
      type: 'delete',
      data: { id: component.id },
      undoData: { type: 'add', component },
      description: description || `删除 ${component.type}`,
    })
  }

  function recordUpdate(
    id: string,
    oldValue: any,
    newValue: any,
    description?: string
  ) {
    pushHistory({
      type: 'update',
      data: { id, updates: newValue },
      undoData: { type: 'update', id, updates: oldValue },
      description: description || '更新组件',
    })
  }

  function recordMove(
    id: string,
    oldPosition: any,
    newPosition: any,
    description?: string
  ) {
    pushHistory({
      type: 'move',
      data: { id, position: newPosition },
      undoData: { type: 'move', id, position: oldPosition },
      description: description || '移动组件',
    })
  }

  function recordBatch(
    operations: Array<{
      type: 'add' | 'update' | 'delete' | 'move'
      data: any
      undoData: any
    }>,
    description?: string
  ) {
    pushHistory({
      type: 'batch',
      data: { operations: operations.map(o => o.data) },
      undoData: {
        type: 'batch',
        operations: operations.map(o => o.undoData).reverse(),
      },
      description: description || '批量操作',
    })
  }

  function clearHistory() {
    store.history = []
    store.historyIndex = 0
  }

  return {
    canUndo,
    canRedo,
    undo,
    redo,
    pushHistory,
    recordAdd,
    recordDelete,
    recordUpdate,
    recordMove,
    recordBatch,
    clearHistory,
  }
}

function generateId(): string {
  return `hist_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`
}
```

```typescript
import { useEditorStore } from '@/stores/editor'

export function useAlignment() {
  const store = useEditorStore()

  const selectedComponents = computed(() => {
    return store.components.filter(c =>
      store.selectionState.componentIds.includes(c.id)
    )
  })

  const getBoundingBox = (components: any[]) => {
    if (components.length === 0) return null

    let minX = Infinity, minY = Infinity
    let maxX = -Infinity, maxY = -Infinity

    components.forEach(comp => {
      if (comp.position.type !== 'absolute') return
      const x = comp.position.x || 0
      const y = comp.position.y || 0
      const w = parseInt(comp.styles.width || '100')
      const h = parseInt(comp.styles.height || '100')

      minX = Math.min(minX, x)
      minY = Math.min(minY, y)
      maxX = Math.max(maxX, x + w)
      maxY = Math.max(maxY, y + h)
    })

    return {
      left: minX,
      top: minY,
      right: maxX,
      bottom: maxY,
      width: maxX - minX,
      height: maxY - minY,
      centerX: (minX + maxX) / 2,
      centerY: (minY + maxY) / 2,
    }
  }

  function alignLeft() {
    const comps = selectedComponents.value.filter(
      c => c.position.type === 'absolute'
    )
    if (comps.length < 2) return

    const bbox = getBoundingBox(comps)
    if (!bbox) return

    const operations = comps.map(comp => {
      const oldPos = { ...comp.position }
      comp.position.x = bbox.left
      return {
        type: 'move' as const,
        data: { id: comp.id, position: { ...comp.position } },
        undoData: { type: 'move' as const, id: comp.id, position: oldPos },
      }
    })

    recordBatch(operations, '左对齐')
  }

  function alignCenterHorizontal() {
    const comps = selectedComponents.value.filter(
      c => c.position.type === 'absolute'
    )
    if (comps.length < 2) return

    if (!bbox) return

    const operations = comps.map(comp => {
      const oldPos = { ...comp.position }
      const w = parseInt(comp.styles.width || '100')

      return {
        type: 'move' as const,
        data: { id: comp.id, position: { ...comp.position } },
        undoData: { type: 'move' as const, id: comp.id, position: oldPos },

      return {

        data: { id: comp.id, position: { ...comp.position } },
        undoData: { type: 'move' as const, id: comp.id, position: oldPos },

    })

    )
    if (comps.length < 2) return

    const bbox = getBoundingBox(comps)
    if (!bbox) return

    const operations = comps.map(comp => {
      const oldPos = { ...comp.position }
      const h = parseInt(comp.styles.height || '100')
      comp.position.y = bbox.centerY - h / 2
      return {
        type: 'move' as const,
        data: { id: comp.id, position: { ...comp.position } },
        undoData: { type: 'move' as const, id: comp.id, position: oldPos },

    recordBatch(operations, '垂直居中对齐')
  }

  function alignBottom() {
    const comps = selectedComponents.value.filter(
      c => c.position.type === 'absolute'
    )
    if (comps.length < 2) return

    const bbox = getBoundingBox(comps)
    if (!bbox) return

    const operations = comps.map(comp => {
      const oldPos = { ...comp.position }
      const h = parseInt(comp.styles.height || '100')
      comp.position.y = bbox.bottom - h
      return {
        type: 'move' as const,
        data: { id: comp.id, position: { ...comp.position } },
        undoData: { type: 'move' as const, id: comp.id, position: oldPos },
      }
    })

    recordBatch(operations, '底部对齐')
  }

  function distributeHorizontal() {
    const comps = selectedComponents.value
      .filter(c => c.position.type === 'absolute')
      .sort((a, b) => (a.position.x || 0) - (b.position.x || 0))

    if (comps.length < 3) return

    const first = comps[0]
    const last = comps[comps.length - 1]
    const firstX = first.position.x || 0
    const lastX = last.position.x || 0
    const lastW = parseInt(last.styles.width || '100')
    const totalSpace = lastX + lastW - firstX

    let totalWidth = 0
    comps.forEach(c => {
      totalWidth += parseInt(c.styles.width || '100')
    })

    const gap = (totalSpace - totalWidth) / (comps.length - 1)
    let currentX = firstX

    const operations = comps.map((comp, idx) => {
      if (idx === 0) return null
      const oldPos = { ...comp.position }
      const prevComp = comps[idx - 1]
      const prevW = parseInt(prevComp.styles.width || '100')
      currentX += prevW + gap
      comp.position.x = currentX
      return {
        type: 'move' as const,
        data: { id: comp.id, position: { ...comp.position } },
        undoData: { type: 'move' as const, id: comp.id, position: oldPos },
      }
    }).filter(Boolean) as any[]

    recordBatch(operations, '水平分布')
  }

  function distributeVertical() {
    const comps = selectedComponents.value
      .filter(c => c.position.type === 'absolute')
      .sort((a, b) => (a.position.y || 0) - (b.position.y || 0))

    if (comps.length < 3) return

    const first = comps[0]
    const last = comps[comps.length - 1]
    const firstY = first.position.y || 0
    const lastY = last.position.y || 0
    const lastH = parseInt(last.styles.height || '100')
    const totalSpace = lastY + lastH - firstY

    let totalHeight = 0
    comps.forEach(c => {
      totalHeight += parseInt(c.styles.height || '100')
    })

    const gap = (totalSpace - totalHeight) / (comps.length - 1)
    let currentY = firstY

    const operations = comps.map((comp, idx) => {
      if (idx === 0) return null
      const oldPos = { ...comp.position }
      const prevComp = comps[idx - 1]
      const prevH = parseInt(prevComp.styles.height || '100')
      currentY += prevH + gap
      comp.position.y = currentY
      return {
        type: 'move' as const,
        data: { id: comp.id, position: { ...comp.position } },
        undoData: { type: 'move' as const, id: comp.id, position: oldPos },
      }
    }).filter(Boolean) as any[]

    recordBatch(operations, '垂直分布')
  }

  return {
    alignLeft,
    alignCenterHorizontal,
    alignRight,
    alignTop,
    alignCenterVertical,
    alignBottom,
    distributeHorizontal,
    distributeVertical,
    getBoundingBox,
  }
}
```

```typescript
// useKeyboardShortcuts.ts
import { onMounted, onUnmounted } from 'vue'
import { useEditorStore } from '@/stores/editor'

type ShortcutHandler = (e: KeyboardEvent) => void

interface Shortcut {
  key: string
  ctrl?: boolean
  shift?: boolean
  alt?: boolean
  meta?: boolean
  handler: ShortcutHandler
  description: string
}

export function useKeyboardShortcuts() {
  const store = useEditorStore()
  const shortcuts: Shortcut[] = []

  function registerShortcut(shortcut: Shortcut) {
    shortcuts.push(shortcut)
  }

  function unregisterShortcut(key: string) {
    const idx = shortcuts.findIndex(s => s.key === key)
    if (idx !== -1) shortcuts.splice(idx, 1)
  }

  function matchesShortcut(e: KeyboardEvent, shortcut: Shortcut): boolean {
    const ctrlKey = shortcut.ctrl ? e.ctrlKey || e.metaKey : !e.ctrlKey && !e.metaKey
    const shiftKey = shortcut.shift ? e.shiftKey : !e.shiftKey
    const altKey = shortcut.alt ? e.altKey : !e.altKey

    return (
      e.key.toLowerCase() === shortcut.key.toLowerCase() &&
      ctrlKey &&
      shiftKey &&
      altKey
    )
  }

  function handleKeyDown(e: KeyboardEvent) {
    // 忽略输入框中的快捷键
    const target = e.target as HTMLElement
    if (
      target.tagName === 'INPUT' ||
      target.tagName === 'TEXTAREA' ||
      target.isContentEditable
    ) {
      // 只允许ESC快捷键
      if (e.key !== 'Escape') return
    }

    for (const shortcut of shortcuts) {
      if (matchesShortcut(e, shortcut)) {
        e.preventDefault()
        e.stopPropagation()
        shortcut.handler(e)
        break
      }
    }
  }

  function registerDefaultShortcuts() {
    // 撤销/重做
    registerShortcut({
      key: 'z',
      ctrl: true,
      handler: () => store.undo(),
      description: '撤销',
    })

    registerShortcut({
      key: 'z',
      ctrl: true,
      shift: true,
      handler: () => store.redo(),
      description: '重做',
    })

    registerShortcut({
      key: 'y',
      ctrl: true,
      handler: () => store.redo(),
      description: '重做',
    })

    // 删除
    registerShortcut({
      key: 'backspace',
      handler: () => {
        if (store.selectedComponentId) {
          store.deleteSelected()
        }
      },
      description: '删除选中',
    })

    registerShortcut({
      key: 'delete',
      handler: () => {
        if (store.selectedComponentId) {
          store.deleteSelected()
        }
      },
      description: '删除选中',
    })

    // 复制/粘贴/剪切
    registerShortcut({
      key: 'c',
      ctrl: true,
      handler: () => {
        if (store.selectedComponentId) {
          store.copyComponent(store.selectedComponentId)
        }
      },
      description: '复制',
    })

    registerShortcut({
      key: 'v',
      ctrl: true,
      handler: () => {
        if (store.clipboard) {
          const newId = store.pasteComponent()
          if (newId) store.selectComponent(newId)
        }
      },
      description: '粘贴',
    })

    registerShortcut({
      key: 'x',
      ctrl: true,
      handler: () => {
        if (store.selectedComponentId) {
          store.cutComponent(store.selectedComponentId)
        }
      },
      description: '剪切',
    })

    // 全选
    registerShortcut({
      key: 'a',
      ctrl: true,
      handler: () => store.selectAll(),
      description: '全选',
    })

    // 取消选择
    registerShortcut({
      key: 'escape',
      handler: () => store.clearSelection(),
      description: '取消选择',
    })

    // 复制组件
    registerShortcut({
      key: 'd',
      ctrl: true,
      handler: () => {
        if (store.selectedComponentId) {
          const newId = store.duplicateComponent(store.selectedComponentId)
          store.selectComponent(newId)
        }
      },
      description: '复制组件',
    })

    // 保存
    registerShortcut({
      key: 's',
      ctrl: true,
      handler: () => store.save(),
      description: '保存',
    })

    // 层级操作
    registerShortcut({
      key: ']',
      ctrl: true,
      handler: () => {
        if (store.selectedComponentId) {
          store.bringToFront(store.selectedComponentId)
        }
      },
      description: '置于顶层',
    })

    registerShortcut({
      key: '[',
      ctrl: true,
      handler: () => {
        if (store.selectedComponentId) {
          store.sendToBack(store.selectedComponentId)
        }
      },
      description: '置于底层',
    })

    // 锁定/解锁
    registerShortcut({
      key: 'l',
      ctrl: true,
      handler: () => {
        if (store.selectedComponentId) {
          store.toggleLock(store.selectedComponentId)
        }
      },
      description: '锁定/解锁',
    })

    // 隐藏/显示
    registerShortcut({
      key: 'h',
      ctrl: true,
      handler: () => {
        if (store.selectedComponentId) {
          store.toggleHidden(store.selectedComponentId)
        }
      },
      description: '隐藏/显示',
    })

    // 缩放
    registerShortcut({
      key: '+',
      ctrl: true,
      handler: () => store.setZoom(Math.min(store.canvasState.zoom + 0.1, 3)),
      description: '放大',
    })

    registerShortcut({
      key: '-',
      ctrl: true,
      handler: () => store.setZoom(Math.max(store.canvasState.zoom - 0.1, 0.1)),
      description: '缩小',
    })

    registerShortcut({
      key: '0',
      ctrl: true,
      handler: () => store.resetViewport(),
      description: '重置视图',
    })

    // 方向键移动
    ;['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight'].forEach(key => {
      registerShortcut({
        key,
        handler: e => {
          if (!store.selectedComponentId) return
          const comp = store.components.find(c => c.id === store.selectedComponentId)
          if (!comp || comp.position.type !== 'absolute') return

          const step = e.shiftKey ? 10 : 1
          const oldPos = JSON.parse(JSON.stringify(comp.position))

          switch (key) {
            case 'ArrowUp':
              comp.position.y = (comp.position.y || 0) - step
              break
            case 'ArrowDown':
              comp.position.y = (comp.position.y || 0) + step
              break
            case 'ArrowLeft':
              comp.position.x = (comp.position.x || 0) - step
              break
            case 'ArrowRight':
              comp.position.x = (comp.position.x || 0) + step
              break
          }

          recordMove(comp.id, oldPos, comp.position)
        },
        description: '移动组件',
      })
    })
  }

  onMounted(() => {
    registerDefaultShortcuts()
    window.addEventListener('keydown', handleKeyDown)
  })

  onUnmounted(() => {
    window.removeEventListener('keydown', handleKeyDown)
  })

  return {
    registerShortcut,
    unregisterShortcut,
  }
}
```

### GET /api/component-templates

**Query 参数:**

```json
  "success": true,
  "data": [
    {
      "id": 1,
      "name": "按钮",
      "category": "基础",
      "component_type": "button",
      "default_props": {},
      "default_styles": {},
      "schema": [],
      "events": [],
      "version": "1.0.0",
      "is_active": true
    }
  ]
}
```

```json
  "success": true,
```

## 边界条件与限制

### 组件限制

1. **最大组件数量:** 单个页面最多 1000 个组件（包括嵌套）
2. **组件嵌套深度:** 最多嵌套 10 层
3. **最小尺寸:** 组件最小宽高 10px
4. **ID 长度限制:** 组件 ID 最多 64 字符

### 历史记录限制

1. **最大历史记录数:** 100 条
2. **单条记录大小:** 最多 1MB
3. **总历史大小:** 最多 50MB

### 画布限制

1. **缩放范围:** 0.1x - 3x
2. **平移范围:** -10000px - 10000px
3. **网格大小:** 8px - 64px (可配置)
4. **吸附阈值:** 5px - 20px (可配置)

### 性能优化

1. **虚拟滚动:** 组件列表超过 100 项时启用
2. **懒渲染:** 视口外组件延迟渲染
3. **节流防抖:** 拖拽、缩放等操作使用节流
4. **增量更新:** 只更新变化的组件

### 数据验证

1. **属性验证:** 按 schema 验证组件属性
2. **样式验证:** 只允许白名单内的样式属性
3. **XSS 防护:** 所有用户输入转义处理
4. **大小限制:** 组件数据单条最多 100KB

### 错误处理

1. **组件加载失败:** 显示占位符，记录日志
2. **历史记录损坏:** 清空历史，继续使用
3. **保存失败:** 提示用户，保留本地草稿
4. **网络错误:** 显示重试按钮，缓存未保存更改
