# 子模块 10: 项目管理与用户权限子模块设计文档

## 概述

负责团队协作、权限控制、项目分享、成员管理等功能。支持多角色权限体系、团队项目协作、项目分享链接等特性。

### 核心功能

- 团队创建与管理
- 成员邀请与角色管理
- 多级权限控制
- 项目分享链接（支持密码保护、有效期）
- 项目导入/导出
- 协作实时同步
- 操作审计日志

```typescript
// 权限矩阵
export const PERMISSION_MATRIX: Record<UserRole, string[]> = {
  owner: ['delete', 'settings', 'edit', 'generate', 'view', 'share', 'add_member', 'remove_member', 'change_role'],
  admin: ['settings', 'edit', 'generate', 'view', 'share', 'add_member', 'remove_member', 'change_role'],
  member: ['edit', 'generate', 'view', 'share'],
  viewer: ['view']
}

export const PROJECT_PERMISSION_LEVELS: Record<ProjectPermission, number> = {
  read: 1,
  write: 2,
  admin: 3
}

// 检查权限
export function hasPermission(role: UserRole, permission: string): boolean {
  return PERMISSION_MATRIX[role]?.includes(permission) ?? false
}

// 检查项目权限等级
export function hasProjectPermission(
  userPermission: ProjectPermission,
  requiredPermission: ProjectPermission
): boolean {
  return PROJECT_PERMISSION_LEVELS[userPermission] >= PROJECT_PERMISSION_LEVELS[requiredPermission]
}
```

### 权限服务

```python
# backend/services/permission_service.py
from typing import Optional
from functools import wraps
from flask import request, g, jsonify

from ..models import (
    Project, Team, TeamMember, UserProjectPermission
)
from .types import UserRole, ProjectPermission


class PermissionService:
    """权限服务"""

    # 角色权限矩阵
    ROLE_PERMISSIONS = {
        UserRole.OWNER: [
            'delete', 'settings', 'edit', 'generate', 'view',
            'share', 'add_member', 'remove_member', 'change_role'
        ],
        UserRole.ADMIN: [
            'settings', 'edit', 'generate', 'view',
            'share', 'add_member', 'remove_member', 'change_role'
        ],
        UserRole.MEMBER: ['edit', 'generate', 'view', 'share'],
        UserRole.VIEWER: ['view']
    }

    # 项目权限等级
    PERMISSION_LEVELS = {
        ProjectPermission.READ: 1,
        ProjectPermission.WRITE: 2,
        ProjectPermission.ADMIN: 3
    }

    @classmethod
    def has_team_permission(cls, user_id: int, team_id: int, permission: str) -> bool:
        """检查团队权限"""
        # 获取用户在团队中的角色
        membership = TeamMember.query.filter_by(
            team_id=team_id,
            user_id=user_id
        ).first()

        if not membership:
            return False

        # 检查权限
        return permission in cls.ROLE_PERMISSIONS.get(membership.role, [])

    @classmethod
    def has_project_permission(
        cls,
        user_id: int,
        project: Project,
        permission: ProjectPermission
    ) -> bool:
        """检查项目权限"""
        # 项目所有者拥有所有权限
        if project.owner_id == user_id:
            return True

        # 检查团队权限
        if project.team_id:
            team_perm = cls._get_team_project_permission(user_id, project.team_id)
            if team_perm and cls._check_permission_level(team_perm, permission):
                return True

        # 检查直接权限
        direct_perm = cls._get_direct_project_permission(user_id, project.id)
        if direct_perm and cls._check_permission_level(direct_perm, permission):
            return True

        return False

    @classmethod
    def _get_team_project_permission(
        cls,
        user_id: int,
        team_id: int
    ) -> Optional[ProjectPermission]:
        """获取用户通过团队获得的项目权限"""
        membership = TeamMember.query.filter_by(
            team_id=team_id,
            user_id=user_id
        ).first()

        if not membership:
            return None

        # 根据角色映射到项目权限
        role_perm_map = {
            UserRole.OWNER: ProjectPermission.ADMIN,
            UserRole.ADMIN: ProjectPermission.ADMIN,
            UserRole.MEMBER: ProjectPermission.WRITE,
            UserRole.VIEWER: ProjectPermission.READ
        }

        return role_perm_map.get(membership.role)

    @classmethod
    def _get_direct_project_permission(
        cls,
        user_id: int,
        project_id: int
    ) -> Optional[ProjectPermission]:
        """获取用户直接的项目权限"""
        perm = UserProjectPermission.query.filter_by(
            user_id=user_id,
            project_id=project_id
        ).first()

        if not perm:
            return None

        return ProjectPermission(perm.permission)

    @classmethod
    def _check_permission_level(
        cls,
        user_permission: ProjectPermission,
        required_permission: ProjectPermission
    ) -> bool:
        """检查权限等级"""
        user_level = cls.PERMISSION_LEVELS.get(user_permission, 0)
        required_level = cls.PERMISSION_LEVELS.get(required_permission, 0)
        return user_level >= required_level

    @classmethod
    def get_user_team_role(cls, user_id: int, team_id: int) -> Optional[UserRole]:
        """获取用户在团队中的角色"""
        membership = TeamMember.query.filter_by(
            team_id=team_id,
            user_id=user_id
        ).first()

        if not membership:
            return None

        return UserRole(membership.role)

            if team.owner_id == user.id:
                g.team = team
                return f(team_id, *args, **kwargs)

            # 检查角色
            user_role = PermissionService.get_user_team_role(user.id, team_id)

            # 检查角色等级
            role_order = [UserRole.VIEWER, UserRole.MEMBER, UserRole.ADMIN, UserRole.OWNER]

                return jsonify({'success': False, 'error': '权限不足'}), 403

            g.team = team
            return f(team_id, *args, **kwargs)

        return decorated_function
    return decorator
```

```python
        expires_days: Optional[int] = None,
        password: Optional[str] = None,
        max_views: Optional[int] = None,
        max_copies: Optional[int] = None
    ) -> ProjectShare:
        """创建分享"""
        token = secrets.token_urlsafe(16)

        expires_at = None
        if expires_days:
            expires_at = datetime.utcnow() + timedelta(days=expires_days)

        password_hash = None
        if password:
            password_hash = cls._hash_password(password)

        share = ProjectShare(
            project_id=project_id,
            share_token=token,
            permission=permission.value,
            expires_at=expires_at,
            is_password_protected=bool(password),
            password_hash=password_hash,
            view_count=0,
            copy_count=0,
            max_views=max_views,
            max_copies=max_copies,
            is_active=True,
            created_by=user_id
        )

        db.session.add(share)
        db.session.commit()

        return share

    @classmethod
    def get_share(
        cls,
        token: str,
        password: Optional[str] = None
    ) -> Optional[ShareInfo]:
        """获取分享信息"""
        share = ProjectShare.query.filter_by(
            share_token=token,
            is_active=True
        ).first()

        if not share:
            return None

        # 检查是否过期
        if share.expires_at and datetime.utcnow() > share.expires_at:
            return None

        # 检查密码
        if share.is_password_protected:
            if not password or not cls._verify_password(password, share.password_hash):
                raise ValueError("密码错误")

        # 检查查看次数
        if share.max_views and share.view_count >= share.max_views:
            raise ValueError("已达最大查看次数")

        # 增加查看计数
        share.view_count += 1
        db.session.commit()

        project = Project.query.get(share.project_id)

        return ShareInfo(
            token=token,
            project=project,
            permission=SharePermission(share.permission),
            expires_at=share.expires_at,
            requires_password=share.is_password_protected,
            can_copy=share.permission in [SharePermission.COPY.value, SharePermission.FORK.value]
        )

    @classmethod
    def copy_project(
        cls,
        share_token: str,
        new_user_id: int,
        new_name: str,
        password: Optional[str] = None
    ) -> Project:
        """复制分享的项目"""
        share = ProjectShare.query.filter_by(
            share_token=share_token,
            is_active=True
        ).first()

        if not share:
            raise ValueError("分享链接不存在")

        # 检查权限
        if share.permission not in [SharePermission.COPY.value, SharePermission.FORK.value]:
            raise ValueError("权限不足")

        # 检查密码
        if share.is_password_protected:
            if not password or not cls._verify_password(password, share.password_hash):
                raise ValueError("密码错误")

        # 检查复制次数
        if share.max_copies and share.copy_count >= share.max_copies:
            raise ValueError("已达最大复制次数")

        # 检查过期
        if share.expires_at and datetime.utcnow() > share.expires_at:
            raise ValueError("分享链接已过期")

        # 复制项目
        old_project = Project.query.get(share.project_id)
        new_project = cls._duplicate_project(old_project, new_user_id, new_name)

        # 增加复制计数
        share.copy_count += 1
        db.session.commit()

        return new_project

    def _duplicate_project(cls, project: Project, new_user_id: int, new_name: str) -> Project:
        """复制项目（需要实现具体逻辑）"""

            thumbnail=project.thumbnail,
            owner_id=new_user_id,
            target_platforms=project.target_platforms,
            is_public=False
        )
        db.session.add(new_project)
        db.session.flush()

        # TODO: 复制页面、组件、数据模型等

        db.session.commit()

    @classmethod
    def _hash_password(cls, password: str) -> str:
        """密码哈希"""
        # 使用简单的哈希，生产环境应使用 bcrypt/argon2
        return hashlib.sha256(password.encode()).hexdigest()

    @classmethod
    def _verify_password(cls, password: str, password_hash: str) -> bool:
        """验证密码"""
        return hmac.compare_digest(
            cls._hash_password(password),
            password_hash
```

```vue
import { ref, reactive, onMounted } from 'vue'
import { ElMessage } from 'element-plus'
import { shareApi } from '@/api/share'

  projectId: number

const showLinkDialog = ref(false)
const creating = ref(false)

  permission: 'view' as SharePermission,
  expires_days: 7,
  password: '',
  max_views: undefined,
  max_copies: undefined

  const res = await shareApi.getShares(props.projectId)

    const res = await shareApi.createShare(createForm)

  }
}

    await shareApi.deleteShare(props.projectId, share.id)

    loadShares()
  } catch (err: any) {
```

```typescript
import request from '@/utils/request'
```
