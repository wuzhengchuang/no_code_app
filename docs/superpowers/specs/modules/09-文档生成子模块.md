# 子模块 09: 文档生成子模块设计文档

## 概述

文档生成子模块负责为通过无代码平台创建的项目自动生成完整的开发文档，包括 CLAUDE.md 规范、技术规格、实现计划、API 文档、目录结构等。确保生成的项目具有一致的代码风格和完整的开发指导。

### 核心功能

- **CLAUDE.md 生成**: 项目专属规范文档，指导 Claude 如何开发该项目
- **技术规格文档**: 详细的技术设计文档
- **实现计划**: 分阶段开发任务清单
- **API 文档**: 接口规格说明
- **目录结构**: 各平台的推荐目录布局
- **README 生成**: 项目说明文档
- **数据库设计文档**: 数据模型说明
- **部署文档**: 部署指南

## 数据模型

### 数据库表设计

```sql
-- 文档生成配置表
CREATE TABLE doc_templates (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL COMMENT '模板名称',
    doc_type VARCHAR(50) NOT NULL COMMENT '文档类型: claude_md, technical_spec, api_spec, readme',
    template_content TEXT NOT NULL COMMENT '模板内容 (Jinja2)',
    description TEXT COMMENT '模板描述',
    is_default BOOLEAN DEFAULT FALSE COMMENT '是否默认模板',
    created_by BIGINT COMMENT '创建者ID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_doc_type (doc_type),
    INDEX idx_is_default (is_default)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='文档模板表';

-- 生成的文档记录表
CREATE TABLE generated_docs (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    project_id BIGINT NOT NULL COMMENT '项目ID',
    doc_type VARCHAR(50) NOT NULL COMMENT '文档类型',
    filename VARCHAR(255) NOT NULL COMMENT '文件名',
    content LONGTEXT COMMENT '文档内容',
    version VARCHAR(20) DEFAULT '1.0' COMMENT '版本号',
    generated_by BIGINT COMMENT '生成者ID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_project (project_id),
    INDEX idx_doc_type (doc_type),
    FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='生成文档记录表';
```

### Python 数据类

```python
from dataclasses import dataclass, field
from typing import List, Dict, Any, Optional
from datetime import datetime
from enum import Enum

class DocType(Enum):
    CLAUDE_MD = "claude_md"
    TECHNICAL_SPEC = "technical_spec"
    IMPLEMENTATION_PLAN = "implementation_plan"
    API_SPEC = "api_spec"
    DIRECTORY_TREE = "directory_tree"
    README = "readme"
    DEPLOYMENT_GUIDE = "deployment_guide"
    DATABASE_DESIGN = "database_design"

@dataclass
class GeneratedDocument:
    id: Optional[int] = None
    filename: str
    content: str
    doc_type: DocType
    format: str = "markdown"
    version: str = "1.0"

@dataclass
class DocTemplate:
    id: Optional[int] = None
    name: str
    doc_type: DocType
    template_content: str
    description: Optional[str] = None
    is_default: bool = False
    created_by: Optional[int] = None
    created_at: Optional[datetime] = None
    updated_at: Optional[datetime] = None

@dataclass
class ProjectContext:
    """用于文档生成的项目上下文"""
    project_id: int
    project_name: str
    description: Optional[str] = None
    target_platforms: List[str] = field(default_factory=list)
    data_models: List[Dict[str, Any]] = field(default_factory=list)
    pages: List[Dict[str, Any]] = field(default_factory=list)
    apis: List[Dict[str, Any]] = field(default_factory=list)
    components: List[Dict[str, Any]] = field(default_factory=list)
    logic_flows: List[Dict[str, Any]] = field(default_factory=list)
    created_at: Optional[datetime] = None
```

## API 接口设计

### 文档生成接口

#### POST /api/documents/generate
生成项目文档

**请求体:**
```json
{
    "project_id": 123,
    "doc_types": [
        "claude_md",
        "technical_spec",
        "implementation_plan",
        "api_spec",
        "readme"
    ],
    "template_overrides": {
        "claude_md": "custom_template_name"
    }
}
```

**响应:**
```json
{
    "success": true,
    "data": {
        "task_id": "doc_gen_abc123",
        "status": "generating",
        "estimated_seconds": 15
    }
}
```

#### GET /api/documents/generate-status/:task_id
查询文档生成状态

**响应:**
```json
{
    "success": true,
    "data": {
        "task_id": "doc_gen_abc123",
        "status": "completed",
        "progress": 100,
```

下载生成的文档

```json
{
    "success": true,
```

**请求体:**
```json
{

    "template_content": "# {{ project_name }}\n...",

```

更新模板

#### DELETE /api/doc-templates/:id
删除模板

## 文档生成器实现

### 核心生成器

```python
from typing import Dict, Any, List, Optional
from dataclasses import dataclass
from jinja2 import Template, Environment, FileSystemLoader, select_autoescape
import os
from datetime import datetime
import json

from .models import DocType, GeneratedDocument, ProjectContext, DocTemplate


class DocumentGenerator:
    """文档生成器核心类"""

    def __init__(self, project_context: ProjectContext, templates_dir: Optional[str] = None):
        self.project = project_context
        self.templates_dir = templates_dir or self._get_default_templates_dir()
        self.env = Environment(
            loader=FileSystemLoader(self.templates_dir),
            autoescape=False,
            trim_blocks=True,
            lstrip_blocks=True
        )
        self._register_filters()

    def _get_default_templates_dir(self) -> str:
        return os.path.join(os.path.dirname(__file__), 'templates')

    def _register_filters(self):
        """注册 Jinja2 过滤器"""
        self.env.filters['tojson'] = lambda v: json.dumps(v, ensure_ascii=False, indent=2)
        self.env.filters['datetime'] = lambda v: v.strftime('%Y-%m-%d %H:%M:%S') if v else ''

    def generate_all(self) -> List[GeneratedDocument]:
        """生成所有文档"""
        documents = []

        # 按顺序生成文档
        documents.append(self.generate_claude_md())
        documents.append(self.generate_readme())
        documents.append(self.generate_technical_spec())
        documents.append(self.generate_implementation_plan())
        documents.append(self.generate_api_spec())
        documents.append(self.generate_database_design())
        documents.append(self.generate_directory_tree())
        documents.append(self.generate_deployment_guide())

        return documents

    def generate_by_type(self, doc_type: DocType) -> GeneratedDocument:
        """按类型生成单个文档"""
        generators = {
            DocType.CLAUDE_MD: self.generate_claude_md,
            DocType.README: self.generate_readme,
            DocType.TECHNICAL_SPEC: self.generate_technical_spec,
            DocType.IMPLEMENTATION_PLAN: self.generate_implementation_plan,
            DocType.API_SPEC: self.generate_api_spec,
            DocType.DATABASE_DESIGN: self.generate_database_design,
            DocType.DIRECTORY_TREE: self.generate_directory_tree,
            DocType.DEPLOYMENT_GUIDE: self.generate_deployment_guide,
        }

        generator = generators.get(doc_type)
        if not generator:
            raise ValueError(f"不支持的文档类型: {doc_type}")

        return generator()

    def generate_claude_md(self) -> GeneratedDocument:
        """生成 CLAUDE.md - 项目专属规范"""
        template = self.env.get_template('claude_md.j2')
        content = template.render(
            project=self.project,
            data_models=self.project.data_models,
            pages=self.project.pages,
            apis=self.project.apis,
            target_platforms=self.project.target_platforms,
            phases=self._generate_phases(),
            now=datetime.utcnow(),
            component_summary=self._summarize_components()
        )
        return GeneratedDocument(
            filename='CLAUDE.md',
            content=content,
            doc_type=DocType.CLAUDE_MD
        )

    def generate_readme(self) -> GeneratedDocument:
        """生成 README.md"""
        template = self.env.get_template('readme.j2')
        content = template.render(
            project=self.project,
            target_platforms=self.project.target_platforms,
            now=datetime.utcnow(),
            quick_start=self._generate_quick_start()
        )
        return GeneratedDocument(
            filename='README.md',
            content=content,
            doc_type=DocType.README
        )

    def generate_technical_spec(self) -> GeneratedDocument:
        """生成技术规格文档"""
        template = self.env.get_template('technical_spec.j2')
        content = template.render(
            project=self.project,
            data_models=self.project.data_models,
            pages=self.project.pages,
            apis=self.project.apis,
            components=self._get_all_components(),
            logic_flows=self.project.logic_flows,
            now=datetime.utcnow()
        )
        return GeneratedDocument(
            filename='TECHNICAL_SPEC.md',
            content=content,
            doc_type=DocType.TECHNICAL_SPEC
        )

    def generate_implementation_plan(self) -> GeneratedDocument:
        """生成实现计划"""
        phases = self._generate_phases()
        template = self.env.get_template('implementation_plan.j2')
        content = template.render(
            project=self.project,
            phases=phases,
            now=datetime.utcnow(),
            estimated_hours=self._estimate_effort(phases)
        )
        return GeneratedDocument(
            filename='IMPLEMENTATION_PLAN.md',
            content=content,
            doc_type=DocType.IMPLEMENTATION_PLAN
        )

    def generate_api_spec(self) -> GeneratedDocument:
        """生成 API 规格文档"""
        template = self.env.get_template('api_spec.j2')
        content = template.render(
            project=self.project,
            apis=self.project.apis,
            now=datetime.utcnow(),
            postman_collection=self._generate_postman_collection()
        )
        return GeneratedDocument(
            filename='API_SPEC.md',
            content=content,
            doc_type=DocType.API_SPEC
        )

    def generate_database_design(self) -> GeneratedDocument:
        """生成数据库设计文档"""
        template = self.env.get_template('database_design.j2')
        content = template.render(
            project=self.project,
            data_models=self.project.data_models,
            now=datetime.utcnow(),
            sql_scripts=self._generate_sql_scripts()
        )
        return GeneratedDocument(
            filename='DATABASE_DESIGN.md',
            content=content,
            doc_type=DocType.DATABASE_DESIGN
        )

    def generate_directory_tree(self) -> GeneratedDocument:
        """生成目录结构文档"""
        trees = []
        for platform in self.project.target_platforms or ['h5']:
            tree = self._generate_platform_tree(platform)
            trees.append((platform, tree))

        template = self.env.get_template('directory_tree.j2')
        content = template.render(
            project=self.project,
            trees=trees,
            now=datetime.utcnow()
        )
        return GeneratedDocument(
            filename='DIRECTORY_TREE.md',
            content=content,
            doc_type=DocType.DIRECTORY_TREE
        )

    def generate_deployment_guide(self) -> GeneratedDocument:
        """生成部署指南"""
        template = self.env.get_template('deployment_guide.j2')
        content = template.render(
            project=self.project,
            target_platforms=self.project.target_platforms,
            now=datetime.utcnow(),
            deploy_steps=self._generate_deploy_steps()
        )
        return GeneratedDocument(
            filename='DEPLOYMENT_GUIDE.md',
            content=content,
            doc_type=DocType.DEPLOYMENT_GUIDE
        )

    def _generate_phases(self) -> List[Dict[str, Any]]:
        """生成实现阶段"""
        phases = []

        # 阶段 1: 项目初始化
        phases.append({
            'number': 1,
            'name': '项目初始化',
            'tasks': [
                '初始化 Git 仓库',
                '配置开发环境',
                '创建项目目录结构',
                '配置代码规范工具 (ESLint, Prettier)',
                '配置 TypeScript',
                '初始化依赖配置 (package.json / requirements.txt)',
                '创建 .gitignore'
            ]
        })

        # 阶段 2: 数据模型
        if self.project.data_models:
            phase2_tasks = []
            for model in self.project.data_models:
                phase2_tasks.append(f'定义 {model.get("name")} 数据模型')
                phase2_tasks.append(f'实现 {model.get("name")} CRUD 操作')

            phases.append({
                'number': 2,
                'name': '数据模型定义',
                'tasks': phase2_tasks
            })

        # 阶段 3: API层
        if self.project.apis:
            phase3_tasks = []
            for api in self.project.apis:
                phase3_tasks.append(f'实现 {api.get("name")} API 接口')

            phases.append({
                'number': 3,
                'name': 'API 层实现',
                'tasks': phase3_tasks
            })

        # 阶段 4: 页面开发
        if self.project.pages:
            phase4_tasks = []
            for page in self.project.pages:
                phase4_tasks.append(f'开发 {page.get("name")} 页面')
                phase4_tasks.append(f'实现 {page.get("name")} 页面样式')

            phases.append({
                'number': 4,
                'name': '页面开发',
                'tasks': phase4_tasks
            })

        # 阶段 5: 逻辑实现
        phases.append({
            'number': 5,
            'name': '业务逻辑实现',
            'tasks': [
                '实现事件处理逻辑',
                '实现状态管理',
                '实现页面导航逻辑',
                '实现数据绑定',
                '实现表单验证',
                '集成 API 调用'
            ]
        })

        # 阶段 6: 测试
        phases.append({
            'number': 6,
            'name': '测试',
            'tasks': [
                '编写单元测试',
                '编写集成测试',
                '编写 E2E 测试',
                '执行完整手动测试',
                'Bug 修复',
                '性能测试与优化'
            ]
        })

        # 阶段 7: 发布
        phases.append({
            'number': 7,
            'name': '发布准备',
            'tasks': [
                '代码评审',
                '性能优化',
                '打包构建',
                '文档完善',
                '部署上线'
            ]
        })

        return phases

    def _get_all_components(self) -> List[Dict[str, Any]]:
        """获取所有页面组件"""
        components = []
        for page in self.project.pages or []:
            page_components = page.get('components', [])
            for comp in page_components:
                comp['_page_name'] = page.get('name')
                comp['_page_path'] = page.get('path')
                components.append(comp)
        return components

    def _summarize_components(self) -> Dict[str, int]:
        """统计组件使用情况"""
        summary = {}
        for comp in self._get_all_components():
            comp_type = comp.get('type', 'unknown')
            summary[comp_type] = summary.get(comp_type, 0) + 1
        return summary

    def _generate_quick_start(self) -> str:
        """生成快速开始指南"""
        platforms = self.project.target_platforms or ['h5']
        guides = []

        for platform in platforms:
            if platform == 'h5':
                guides.append('''
### H5 开发

\`\`\`bash
# 安装依赖
npm install

# 启动开发服务器
npm run dev

# 构建生产版本
npm run build
\`\`\`
''')
            elif platform == 'wechat_miniapp':
                guides.append('''
### 微信小程序开发

1. 下载并安装 [微信开发者工具](https://developers.weixin.qq.com/miniprogram/dev/devtools/download.html)

            elif platform == 'flutter':
                guides.append('''

flutter pub get

flutter run

        return '\n'.join(guides)

    def _generate_postman_collection(self) -> Dict[str, Any]:
        """生成 Postman 集合"""
        if not self.project.apis:
            return {}

        collection = {
            "info": {
                "name": f"{self.project.project_name} API",
                "schema": "https://schema.getpostman.com/json/collection/v2.1.0/collection.json"
            },
            "item": []
        }

        for api in self.project.apis:
            item = {
                "name": api.get("name"),
                "request": {
                    "method": api.get("method", "GET").upper(),
                    "header": api.get("headers", []),
                    "url": {
                        "raw": api.get("url"),
                        "host": ["{{base_url}}"],
                        "path": api.get("url", "").lstrip("/").split("/")
                    }
                }
            }
            collection["item"].append(item)

        return collection

    def _generate_sql_scripts(self) -> List[Dict[str, Any]]:
        """生成 SQL 脚本"""
        scripts = []

        for model in self.project.data_models:
            table_name = model.get("table_name", model.get("name", "").lower())
            fields = model.get("fields", [])

            sql_lines = [f"CREATE TABLE IF NOT EXISTS `{table_name}` ("]
            sql_lines.append("  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,")

            for field in fields:
                field_name = field.get("name", "").split(".")[-1]
                field_type = self._map_to_mysql_type(field.get("type", "string"))
                nullable = "NULL" if not field.get("is_required") else "NOT NULL"
                default = f" DEFAULT {field.get('default')}" if field.get("default") else ""
                comment = f" COMMENT '{field.get('description', '')}'"

                sql_lines.append(f"  `{field_name}` {field_type} {nullable}{default}{comment},")

            sql_lines.append("  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,")
            sql_lines.append("  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP")
            sql_lines.append(") ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;")

            scripts.append({
                "model_name": model.get("name"),
                "sql": "\n".join(sql_lines)
            })

        return scripts

    def _map_to_mysql_type(self, field_type: str) -> str:
        """字段类型映射到 MySQL 类型"""
        type_mapping = {
            "string": "VARCHAR(255)",
            "text": "TEXT",
            "number": "INT",
            "float": "DECIMAL(10,2)",
            "boolean": "BOOLEAN",
            "date": "DATE",
            "datetime": "DATETIME",
            "json": "JSON",
            "array": "JSON"
        }
        return type_mapping.get(field_type.lower(), "VARCHAR(255)")

    def _estimate_effort(self, phases: List[Dict[str, Any]]) -> Dict[str, int]:
        """估算工作量"""
        total_tasks = sum(len(phase.get('tasks', [])) for phase in phases)
        return {
            "total_tasks": total_tasks,
            "estimated_hours": total_tasks * 4,  # 每个任务平均 4 小时
            "phases": len(phases)
        }

    def _generate_deploy_steps(self) -> Dict[str, List[str]]:
        """生成各平台部署步骤"""
        steps = {}

        for platform in self.project.target_platforms or ['h5']:
```

```python
from typing import List, Optional
from datetime import datetime
import uuid
import asyncio

from .models import (
    ProjectContext, GeneratedDocument, DocType, DocTemplate
)
from .generator import DocumentGenerator
from ..extensions import db
from ..models import GeneratedDoc as GeneratedDocModel, DocTemplate as DocTemplateModel


class DocumentService:
    """文档服务"""

    def __init__(self):
        self.task_status = {}

    async def generate_documents(
        self,
        project_id: int,
        doc_types: List[str],
        user_id: Optional[int] = None
    ) -> str:
        """异步生成文档"""
        # 创建任务
        task_id = f"doc_gen_{uuid.uuid4().hex[:8]}"
        self.task_status[task_id] = {
            'status': 'pending',
            'progress': 0,
            'documents': []
        }

        # 异步执行生成
        asyncio.create_task(
            self._generate_async(task_id, project_id, doc_types, user_id)
        )

        return task_id

    async def _generate_async(
        self,
        task_id: str,
        project_id: int,
        doc_types: List[str],
        user_id: Optional[int]
    ):
        """异步生成任务"""
        try:
            self.task_status[task_id]['status'] = 'generating'
            self.task_status[task_id]['progress'] = 10

            # 获取项目上下文
            project_context = await self._get_project_context(project_id)

            self.task_status[task_id]['progress'] = 30

            # 创建生成器
            generator = DocumentGenerator(project_context)

            self.task_status[task_id]['progress'] = 50

            # 生成文档
            generated_docs = []
            total = len(doc_types)

            for i, doc_type_str in enumerate(doc_types):
                doc_type = DocType(doc_type_str)
                doc = generator.generate_by_type(doc_type)

                # 保存到数据库
                doc_model = GeneratedDocModel(
                    project_id=project_id,
                    doc_type=doc_type.value,
                    filename=doc.filename,
                    content=doc.content,
                    version=doc.version,
                    generated_by=user_id
                )
                db.session.add(doc_model)

                generated_docs.append({
                    'doc_type': doc_type.value,
                    'filename': doc.filename,
                    'download_url': f'/api/documents/download/{doc_model.id}'
                })

                self.task_status[task_id]['progress'] = 50 + int((i + 1) / total * 40)

            db.session.commit()

            self.task_status[task_id]['status'] = 'completed'
            self.task_status[task_id]['progress'] = 100
            self.task_status[task_id]['documents'] = generated_docs

        except Exception as e:
            self.task_status[task_id]['status'] = 'failed'
            self.task_status[task_id]['error'] = str(e)
            db.session.rollback()

    def get_task_status(self, task_id: str) -> Optional[dict]:
        """获取任务状态"""
        return self.task_status.get(task_id)

        # 获取数据模型
        data_models = []
        # ... 从数据库加载

        pages = []
        # ... 从数据库加载

        # 获取 API
        apis = []
        # ... 从数据库加载

        return ProjectContext(
            project_id=project.id,
```

```python
    created_at = db.Column(db.DateTime, nullable=False, default=datetime.utcnow)

    project = db.relationship('Project', backref=db.backref('generated_docs', cascade='all, delete-orphan'))
```

### API 路由

```python
# backend/routes/documents.py
from flask import Blueprint, request, jsonify, send_file
from io import BytesIO
from ..services.document_service import DocumentService
from ..models import GeneratedDoc as GeneratedDocModel
from ..extensions import db

documents_bp = Blueprint('documents', __name__)
doc_service = DocumentService()


@documents_bp.route('/generate', methods=['POST'])
def generate():
    """生成文档"""
    data = request.get_json()
    project_id = data.get('project_id')
    doc_types = data.get('doc_types', [])

    task_id = doc_service.generate_documents(project_id, doc_types)

    return jsonify({
        'success': True,
        'data': {
            'task_id': task_id,
            'status': 'generating',
            'estimated_seconds': 15
        }
    })


@documents_bp.route('/generate-status/<task_id>', methods=['GET'])
def get_generate_status(task_id):
    """获取生成状态"""
    status = doc_service.get_task_status(task_id)
    if not status:
        return jsonify({'success': False, 'error': '任务不存在'}), 404

    return jsonify({
        'success': True,
        'data': status
    })


@documents_bp.route('/download/<int:doc_id>', methods=['GET'])
def download(doc_id):
    """下载文档"""
    doc = GeneratedDocModel.query.get_or_404(doc_id)

    buffer = BytesIO(doc.content.encode('utf-8'))
    buffer.seek(0)

    return send_file(
        buffer,
        as_attachment=True,
        download_name=doc.filename,
        mimetype='text/markdown'
    )


@documents_bp.route('/templates', methods=['GET'])
def list_templates():
    """获取模板列表"""
    from ..models import DocTemplate

    doc_type = request.args.get('doc_type')
    query = DocTemplate.query

    if doc_type:
        query = query.filter_by(doc_type=doc_type)

    templates = query.order_by(DocTemplate.is_default.desc()).all()

    return jsonify({
        'success': True,
        'data': [
            {
                'id': t.id,
                'name': t.name,
                'doc_type': t.doc_type,
                'is_default': t.is_default,
                'description': t.description,
                'created_at': t.created_at.isoformat()
            }
            for t in templates
        ]
    })
```

## 设计规范

### 文档命名规范

- 主文档使用大驼峰命名: `CLAUDE.md`, `README.md`
- 其他文档使用下划线分隔: `TECHNICAL_SPEC.md`, `IMPLEMENTATION_PLAN.md`
- 所有文档使用 `.md` 后缀

### 模板规范

- 模板使用 Jinja2 语法
- 模板文件命名: `{doc_type}.j2`
- 模板中使用的变量必须在 ProjectContext 中定义
- 模板注释使用 `{# ... #}` 格式
