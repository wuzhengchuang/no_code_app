### 技术栈

| 层级 | 技术选型 |
|------|----------|
| 前端 | Vue 3 + TypeScript + Tailwind CSS |
| 后端服务 | Python |
| 服务网关 | Node.js |
| 数据库 | MySQL |
| 桌面应用 | Electron + Node.js |

### 核心子模块清单

整个平台划分为以下独立子模块：

1. **用户与权限子模块** - 用户注册登录、权限管理、团队协作
2. **项目管理子模块** - 项目创建、编辑、删除、分享、导入导出
3. **可视化编辑器子模块** - 拖拽组件、布局编排、属性配置
4. **组件库子模块** - 预设组件的定义、分类、版本管理
5. **接口配置子模块** - 接口配置、JSON解析、数据模型生成
6. **逻辑编排子模块** - 事件-动作绑定、数据流配置
7. **App预览子模块** - 实时预览、交互模拟、真实接口调用
8. **大模型集成子模块** - LLM抽象、提示词管理、代码生成请求
9. **代码生成引擎子模块** - 插件架构、各平台代码生成器
10. **文档生成子模块** - CLAUDE.md格式、项目规范、实现计划
11. **桌面应用子模块** - Electron封装、离线功能

### 3.2 核心数据结构

```typescript
// 拖拽状态
interface DragState {
  isDragging: boolean
  componentType?: string  // 从组件库拖入时
  componentId?: string    // 已有组件移动时
  startX: number
  startY: number
  currentX: number
  currentY: number
}

// 组件在画布上的位置
interface ComponentPosition {
  type: 'absolute' | 'relative' | 'flow'  // 绝对定位/相对定位/流式布局
  x?: number      // 绝对定位X
  y?: number      // 绝对定位Y
  width?: string  // '100%' | '200px'
  height?: string
  flex?: number   // flex布局权重
  gridColumn?: string
  gridRow?: string
}

// 组件定义
interface ComponentDefinition {
  type: string                    // 组件类型标识
  name: string                    // 显示名称
  category: string                // 分类
  icon: string                    // 图标
  description: string

  // 属性Schema - 用于生成右侧属性面板表单
  propsSchema: PropSchema[]

  // 默认属性值
  defaultProps: Record<string, any>

  // 事件列表
  events: string[]

  // 插槽支持（用于嵌套组件）
  slots?: SlotDefinition[]
}

// 属性Schema
interface PropSchema {
  name: string
  label: string
  type: 'string' | 'number' | 'boolean' | 'select' | 'color' | 'image' | 'json'
  default?: any
  options?: { label: string; value: any }[]  // select类型用
  required?: boolean
  placeholder?: string
  group?: string  // 属性分组: '基础' | '样式' | '高级'
}

// 项目状态管理 (Pinia Store)
interface EditorStore {
  project: Project
  currentPageId: string
  selectedComponentId: string | null
  clipboard: ComponentData | null
  history: HistoryEntry[]
  historyIndex: number

  // 操作方法
  selectComponent(id: string | null): void
  addComponent(component: ComponentData): void
  updateComponent(id: string, updates: Partial<ComponentData>): void
  deleteComponent(id: string): void
  moveComponent(id: string, position: ComponentPosition): void
  undo(): void
  redo(): void
}
```

### 4.2 JSON解析与数据模型生成算法

```python
def json_to_data_model(json_example: dict, model_name: str) -> DataModel:
    """
    将JSON示例转换为数据模型定义
    """
    fields = []

    def parse_value(path: str, value: Any):
        """递归解析JSON值"""
        if value is None:
            fields.append({
                'name': path,
                'type': 'string',
                'is_nullable': True
            })
        elif isinstance(value, bool):
            fields.append({'name': path, 'type': 'boolean'})
        elif isinstance(value, int):
            fields.append({'name': path, 'type': 'number'})
        elif isinstance(value, float):
            fields.append({'name': path, 'type': 'number'})
        elif isinstance(value, str):
            if is_date_format(value):
                fields.append({'name': path, 'type': 'date'})
            elif is_url_format(value):
                fields.append({'name': path, 'type': 'url'})
            else:
                fields.append({'name': path, 'type': 'string'})
        elif isinstance(value, list):
            fields.append({'name': path, 'type': 'array'})
            if value and len(value) > 0:
                parse_value(f'{path}[]', value[0])
        elif isinstance(value, dict):
            fields.append({'name': path, 'type': 'object'})
            for key, val in value.items():
                parse_value(f'{path}.{key}' if path else key, val)

    parse_value('', json_example)

    return DataModel(
        name=model_name,
        fields=fields,
        source='from_api'
    )
```

```typescript
class PreviewEngine {
  project: Project
  currentPage: Page | null
  state: Record<string, any> = {}
  apiMocks: Map<string, MockResponse>

  constructor(project: Project) {
    this.project = project
  }

  async loadPage(pageId: string): Promise<void> {
    this.currentPage = this.project.pages.find(p => p.id === pageId)
    this.initPageState()
    await this.triggerPageLoad()
  }

  initPageState(): void {
    for (const stateDef of this.currentPage.states) {
      this.state[stateDef.name] = stateDef.defaultValue
    }
  }

  async triggerPageLoad(): Promise<void> {
    const loadBindings = this.project.eventBindings.filter(
      b => b.componentId === this.currentPage.id && b.eventName === 'load'
    )
    for (const binding of loadBindings) {
      await this.executeActions(binding.actions)
    }
  }

  async executeActions(actions: Action[]): Promise<void> {
    for (const action of actions) {
      if (action.condition && !this.evaluateCondition(action.condition)) {
        continue
      }
      await this.executeAction(action)
    }
  }

  async callRealApi(apiConfig: ApiConfig, params: Record<string, any>): Promise<any> {
    const headers = this.parseHeaders(apiConfig.headers)
    const url = this.interpolateUrl(apiConfig.url, params)

    const response = await fetch(url, {
      method: apiConfig.method,
      headers,
      body: apiConfig.method !== 'GET' ? JSON.stringify(params) : undefined
    })

    return await response.json()
  }
}
```
